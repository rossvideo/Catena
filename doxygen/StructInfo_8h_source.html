176template <CatenaStruct T>
177void toProto(catena::Value& dst, const T* src, const ParamDescriptor& pd, const Authorizer& authz) {
178
179 auto fields = StructInfo<T>::fields; // get tuple of FieldInfo for this struct type
180 auto* dstFields = dst.mutable_struct_value()->mutable_fields();
181
182 // lambda function to call toProto for the given field if it is authorized
183 auto readField = [&](const auto& field) {
184 ParamDescriptor& subParam = pd.getSubParam(field.name);
185 if (authz.readAuthz(subParam)) {
186 catena::Value* newFieldValue = (*dstFields)[field.name].mutable_value();
187
195 toProto(*newFieldValue, &(src->*(field.memberPtr)), subParam, authz);
196 }
197 };
198
199 // call readField for each field in the struct
200 std::apply([&](auto... field) {
201 (readField(field), ...);
202 }, fields);
203}

214template <meta::IsVariant T>
215void toProto(catena::Value& dst, const T* src, const ParamDescriptor& pd, const Authorizer& authz) {
216 std::string variantType = alternativeNames<T>[src->index()];
217 ParamDescriptor& subParam = pd.getSubParam(variantType);
218 std::visit([&](auto& arg) {
219 catena::Value elemValue;
220 toProto(elemValue, &arg, subParam, authz);
221 catena::StructVariantValue* structVariant = dst.mutable_struct_variant_value();
222 structVariant->set_struct_variant_type(variantType);
223 *structVariant->mutable_value() = elemValue;
224 }, *src);

244template <CatenaStructArray T>
245void fromProto(const catena::Value& src, T* dst, const ParamDescriptor& pd, const Authorizer& authz) {
246 using structType = T::value_type;
247 if (!src.has_struct_array_values()) {
248 // src is not an array so it cannot be deserialized into an array
249 return;
250 }
251 auto& srcArray = src.struct_array_values().struct_values();
252 dst->clear(); // empty the destination vector
253
254 // Right now from proto is able to append any number of values to the vector
255 // Do we want to keep this behavior?
256
257 // iterate over each element in the src array and call fromProto for each
258 for (int i = 0; i < srcArray.size(); ++i) {
259 catena::Value item;
260 *item.mutable_struct_value() = srcArray.Get(i);
261 structType& elemValue = dst->emplace_back();
262 fromProto(item, &elemValue, pd, authz);
263 }

275template <CatenaStruct T>
276void fromProto(const catena::Value& src, T* dst, const ParamDescriptor& pd, const Authorizer& authz) {
277 auto fields = StructInfo<T>::fields; // get tuple of FieldInfo for this struct type
278 if (!src.has_struct_value()) {
279 // src is not a struct so it cannot be deserialized into a struct
280 return;
281 }
282 auto& srcFields = src.struct_value().fields();
283
284 // lambda function to call fromProto for the given field if it is authorized
285 auto writeField = [&](const auto& field) {
286 ParamDescriptor& subParam = pd.getSubParam(field.name);
287 if (!authz.writeAuthz(subParam)) {
288 // not authorized to write to this field or the field is not in the src
289 return;
290 }
291 if (!srcFields.contains(field.name)) {
292 // field not found in src so it is not being updated
293 return;
294 }
295 const catena::Value& fieldValue = srcFields.at(field.name).value();
296
304 fromProto(fieldValue, &(dst->*(field.memberPtr)), subParam, authz);
305 };
306
307 // call writeField for each field in the dst struct
308 // If the src value contains a field that is not in the dst struct, it will be ignored
309 std::apply([&](auto... field) {
310 (writeField(field), ...);
311 }, fields);
312}

356template <meta::IsVariant V, std::size_t Index>
357void _changeType(V& variant, const std::size_t newTypeIndex) {
358 if constexpr (Index == std::variant_size_v<V>) {
359 // reached the end of the variant, type not found
360 return;
361 } else {
362 using NewType = std::variant_alternative_t<Index, V>;
363 if (newTypeIndex == Index) {
364 // found the new type, change the variant to the default value of the new type
365 variant = NewType{};
366 } else {
367 _changeType<V, Index + 1>(variant, newTypeIndex);
368 }
369 }
370}

372template <meta::IsVariant T>
373void fromProto(const catena::Value& src, T* dst, const ParamDescriptor& pd, const Authorizer& authz) {
374 if (!src.has_struct_variant_value()) {
375 // src is not a variant so it cannot be deserialized into a variant
376 return;
377 }
378 const catena::StructVariantValue& srcVariant = src.struct_variant_value();
379 std::string variantType = srcVariant.struct_variant_type();
380
381 std::size_t typeIndex = _findTypeIndex(variantType, alternativeNames<T>);
382 if (typeIndex >= alternativeNames<T>.size()) {
383 // variant type not found in the list of alternatives
384 return;
385 }
386 if (typeIndex != dst->index()) {
387 // The type of the variant needs to be changed
388 _changeType<T, 0>(*dst, typeIndex);
389 }
390
391 ParamDescriptor& subParam= pd.getSubParam(variantType);
392 std::visit([&](auto& arg) {
393 fromProto(srcVariant.value(), &arg, subParam, authz);
394 }, *dst);
395}
