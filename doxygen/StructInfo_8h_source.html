154 template <CatenaStruct T>
155 void toProto(catena::Value& dst, const T* src, const AuthzInfo& auth) {
156 
157  auto fields = StructInfo<T>::fields; // get tuple of FieldInfo for this struct type
158  auto* dstFields = dst.mutable_struct_value()->mutable_fields();
159 
160  // lambda function to call toProto for the given field if it is authorized
161  auto readField = [&](const auto& field) {
162  AuthzInfo subParamAuth = auth.subParamInfo(field.name);
163  if (subParamAuth.readAuthz()) {
164  catena::Value* newFieldValue = (*dstFields)[field.name].mutable_value();

200 template <CatenaStructArray T>
201 void fromProto(const catena::Value& src, T* dst, const AuthzInfo& auth) {
202  using structType = T::value_type;
203  if (!src.has_struct_array_values()) {
204  // src is not an array so it cannot be deserialized into an array
205  return;
206  }
207  auto& srcArray = src.struct_array_values().struct_values();
208  dst->clear(); // empty the destination vector
209 
210  // Right now from proto is able to append any number of values to the vector
211  // Do we want to keep this behavior?
212 
213  // iterate over each element in the src array and call fromProto for each
214  for (int i = 0; i < srcArray.size(); ++i) {
215  catena::Value item;
216  *item.mutable_struct_value() = srcArray.Get(i);
217  structType& elemValue = dst->emplace_back();
218  fromProto(item, &elemValue, auth);
219  }
220 }

231 template <CatenaStruct T>
232 void fromProto(const catena::Value& src, T* dst, const AuthzInfo& auth) {
233  auto fields = StructInfo<T>::fields; // get tuple of FieldInfo for this struct type
234  if (!src.has_struct_value()) {
235  // src is not a struct so it cannot be deserialized into a struct
236  return;
237  }
238  auto& srcFields = src.struct_value().fields();
239 
240  // lambda function to call fromProto for the given field if it is authorized
241  auto writeField = [&](const auto& field) {
242  AuthzInfo subParamAuth = auth.subParamInfo(field.name);
243  if (!subParamAuth.writeAuthz()) {
244  // not authorized to write to this field or the field is not in the src
245  return;
246  }
247  if (!srcFields.contains(field.name)) {
248  // field not found in src so it is not being updated
249  return;
250  }
251  const catena::Value& fieldValue = srcFields.at(field.name).value();
252 
260  fromProto(fieldValue, &(dst->*(field.memberPtr)), subParamAuth);
261  };
262 
263  // call writeField for each field in the dst struct
264  // If the src value contains a field that is not in the dst struct, it will be ignored
265  std::apply([&](auto... field) {
266  (writeField(field), ...);
267  }, fields);
268 }
