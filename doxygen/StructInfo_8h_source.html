222  auto fields = StructInfo<T>::fields; // get tuple of FieldInfo for this struct type
223  if (!src.has_struct_value()) {

229  // lambda function to call fromProto for the given field if it is authorized

231  AuthzInfo subParamAuth = auth.subParamInfo(field.name);

234  return;
