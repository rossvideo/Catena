81template <typename E> class EnumDecorator {
82 // cause compile time error if E is not an enum
83 static_assert(std::is_enum<E>::value, "EnumDecorator requires an enum type");
84
85 public:
86 using etype = E; /*< template's enum parameter */
87 using utype = typename std::underlying_type<E>::type; /*< underlying base type */
88 using FwdMap = std::unordered_map<E, std::string>; /*< forward map of enum values to strings */
89 using RevMap = std::unordered_map<std::string, E>; /*< reverse map of strings to enum values */

102 EnumDecorator(E value) : value_{value} {}

113 EnumDecorator(const std::string& str) : value_{static_cast<E>(0)} {
114 const RevMap& revMap = getReverseMap();
115 if (revMap.contains(str)) {
116 value_ = revMap.at(str);
117 }
118 }
119
127 const RevMap& getReverseMap() {
128 static RevMap revMap;
129 if (revMap.empty()) {
130 for (const auto& pair : fwdMap_) {
131 revMap[pair.second] = pair.first;
132 }
133 }
134 return revMap;
135 }
136
142 const FwdMap& getForwardMap() const { return fwdMap_; }
143
151 EnumDecorator(utype value) : value_{static_cast<E>(value)} {
152 if (!fwdMap_.contains(value_)) {
153 value_ = static_cast<E>(0);
154 }
155 }

199 return std::string();
200 }
201 }

206 const std::string& toString() const {
207 if (fwdMap_.contains(value_)) {
208 return fwdMap_.at(value_);

210 static std::string empty;
