120 bool addProduct(const K key, Maker maker) {
121 LockGuard lock(_mtx);
122 if (!_registry.contains(key)) {
123 _registry[key] = maker;
124 } else {
125 std::stringstream err;
126 err << __PRETTY_FUNCTION__;
127 err << ", attempted to register item with duplicate key";
128 catena::meta::stream_if_possible(err, key);
129 throw std::runtime_error(err.str());
130 }
131 return true;
132 }

143 std::unique_ptr<P> makeProduct(const K key, Ms&&... args) {
144 LockGuard lock(_mtx);
145 if (_registry.contains(key)) {
146 std::unique_ptr<P> ans;
147 try {
148 // N.B. P could be a base class with pure virtual methods, so
149 // we can't use std::make_unique here.
150 P* p = _registry.at(key)(std::forward<Ms>(args)...);
151 ans.reset(p);
152 return ans;
153 }
154 catch (std::exception& e) {
155 // in case the product maker throws an exception
156 std::stringstream err;
157 err << __PRETTY_FUNCTION__;
158 err << ", caught exception: ";
159 err << e.what();
160 throw std::runtime_error(err.str());
161 }
162 } else {
163 std::stringstream err;
164 err << __PRETTY_FUNCTION__;
165 err << ", could not find entry with key: ";
166 catena::meta::stream_if_possible(err, key);
167 throw std::runtime_error(err.str());
168 }
169 }

179 bool canMake(const K key) const {
