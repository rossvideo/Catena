129 bool satisfied(const catena::Value& src) const override {
130 bool ans = false;
131 if constexpr(std::is_same<T, int32_t>::value) {
132 ans = src.int32_value() >= min_
133 && src.int32_value() <= max_
134 && (!step_ || (src.int32_value() - min_) % step_ == 0);
135 }
136 else if constexpr(std::is_same<T, float>::value) {
137 ans = src.float32_value() >= min_
138 && src.float32_value() <= max_
139 && (!step_ || std::fmod(src.float32_value() - min_, step_) == 0);
140 }
141 return ans;
142 }
143
153 catena::Value apply(const catena::Value& src) const override {
154 catena::Value constrainedVal;
155 if constexpr(std::is_same<T, int32_t>::value) {
156 // return empty value if src is not valid
157 if (src.has_int32_value()) {
158 int32_t s = src.int32_value();
159 // constrain if not within allowed range
160 if (s < min_) {
161 constrainedVal.set_int32_value(min_);
162 } else if (s > max_) {
163 constrainedVal.set_int32_value(max_);
164 } else if (step_ && (s - min_) % step_ != 0) {
165 constrainedVal.set_int32_value(s - ((s - min_) % step_));
166 } else {
167 constrainedVal.set_int32_value(s);
168 }
169 }
170 }
171 else if constexpr(std::is_same<T, float>::value) {
172 // return empty value if src is not valid
173 if (src.has_float32_value()) {
174 float s = src.float32_value();
175 // constrain if not within allowed range
176 if (s < min_) {
177 constrainedVal.set_float32_value(min_);
178 } else if (s > max_) {
179 constrainedVal.set_float32_value(max_);
180 } else if (step_ && std::fmod(s - min_, step_) != 0) {
181 constrainedVal.set_float32_value(s - std::fmod(s - min_, step_));
182 } else {
183 constrainedVal.set_float32_value(s);
184 }
185 }
186 }
187 return constrainedVal;
188 }
