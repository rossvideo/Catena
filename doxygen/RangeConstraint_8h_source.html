70 RangeConstraint(T min, T max, T step, T display_min, T display_max, std::string oid, bool shared)
71 : min_(min), max_(max), display_min_{display_min},
72 display_max_{display_max}, oid_{oid}, shared_{shared} {
73 if constexpr(!std::is_same<T, int32_t>::value && !std::is_same<T, float>::value) {
74 throw std::runtime_error("Cannot create RangeConstraint with type other than int32_t or float");
75 }
76 // want step to always be positive
77 if constexpr (std::is_same<T, int32_t>::value) {
78 step_ = std::abs(step);
79 } else if constexpr (std::is_same<T, float>::value) {
80 step_ = std::fabs(step);
81 }
82 }

135 bool satisfied(const st2138::Value& src) const override {
136 bool ans = false;
137 if constexpr(std::is_same<T, int32_t>::value) {
138 ans = src.has_int32_value() && src.int32_value() >= min_
139 && src.int32_value() <= max_
140 && (!step_ || (src.int32_value() - min_) % step_ == 0);
141 }
142 else if constexpr(std::is_same<T, float>::value) {
143 if (src.has_float32_value()) {
144 const float val = src.float32_value();
145 if (val >= min_ && val <= max_) {
146 if (!step_) {
147 ans = true;
148 } else {
149 constexpr float kStepTolerance = 0.0001f;
150 const st2138::Value constrainedVal = apply(src);
151 // close enough if within 1/10000th of the step size
152 ans = std::fabs(constrainedVal.float32_value() - val) < (step_ * kStepTolerance);
153 }
154 }
155 }
156 }
157 return ans;
158 }
159
169 st2138::Value apply(const st2138::Value& src) const override {
170 st2138::Value constrainedVal;
171 if constexpr(std::is_same<T, int32_t>::value) {
172 // return empty value if src is not valid
173 if (src.has_int32_value()) {
174 int32_t s = src.int32_value();
175 // constrain if not within allowed range
176 if (s < min_) {
177 constrainedVal.set_int32_value(min_);
178 } else if (s > max_) {
179 constrainedVal.set_int32_value(max_);
180 } else if (step_ && (s - min_) % step_ != 0) {
181 constrainedVal.set_int32_value(s - ((s - min_) % step_));
182 } else {
183 constrainedVal.set_int32_value(s);
184 }
185 }
186 }
187 else if constexpr(std::is_same<T, float>::value) {
188 // return empty value if src is not valid
189 if (src.has_float32_value()) {
190 float val = src.float32_value();
191 // constrain if not within allowed range
192 if (val < min_) {
193 constrainedVal.set_float32_value(min_);
194 } else if (val > max_) {
195 constrainedVal.set_float32_value(max_);
196 } else if (step_) {
197 // round to the nearest step, if step is not 0
198 const float steps = std::round((val - min_) / step_);
199 const float newVal = (steps * step_) + min_;
200 // make sure we don't go under the min or over the max when rounding
201 if (newVal < min_) {
202 constrainedVal.set_float32_value(min_);
203 } else if (newVal > max_) {
204 constrainedVal.set_float32_value(max_);
205 } else {
206 constrainedVal.set_float32_value(newVal);
207 }
208 } else {
209 constrainedVal.set_float32_value(val);
210 }
211 }
212 }
213 return constrainedVal;
214 }
