138 bool satisfied(const catena::Value& src) const override {
139
140 if constexpr(std::is_same<T, int32_t>::value) {
141 return src.int32_value() >= min_
142 && src.int32_value() <= max_
143 && (!step_ || (src.int32_value() - min_) % step_ == 0);
144 }
145
146 if constexpr(std::is_same<T, float>::value) {
147 return src.float32_value() >= min_
148 && src.float32_value() <= max_
149 && (!step_ || std::fmod(src.float32_value() - min_, step_) == 0);
150 }
151 }
152
162 catena::Value apply(const catena::Value& src) const override {
163 catena::Value constrainedVal;
164
165 if constexpr(std::is_same<T, int32_t>::value) {
166 // return empty value if src is not valid
167 if (!src.has_int32_value()) { return constrainedVal; }
168 int32_t s = src.int32_value();
169
170 // constrain if not within allowed range
171 if (s < min_) {
172 constrainedVal.set_int32_value(min_);
173 } else if (s > max_) {
174 constrainedVal.set_int32_value(max_);
175 } else if (!step_ || (s - min_) % step_ != 0) {
176 constrainedVal.set_int32_value(s - ((s - min_) % step_));
177 }
178 return constrainedVal;
179 }
180
181 if constexpr(std::is_same<T, float>::value) {
182
183 // return empty value if src is not valid
184 if (!src.has_float32_value()) { return constrainedVal; }
185 float s = src.float32_value();
186
187 // constrain if not within allowed range
188 if (s < min_) {
189 constrainedVal.set_float32_value(min_);
190 } else if (s > max_) {
191 constrainedVal.set_float32_value(max_);
192 } else if (!step_ || std::fmod(s - min_, step_) != 0) {
193 constrainedVal.set_float32_value(s - std::fmod(s - min_, step_));
194 }
195 return constrainedVal;
196 }
197 }
