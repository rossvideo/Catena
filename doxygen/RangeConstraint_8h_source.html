162  catena::Value apply(const catena::Value& src) const override {
163  catena::Value constrainedVal;
164 
165  if constexpr(std::is_same<T, int32_t>::value) {
166  // return empty value if src is not valid
167  if (!src.has_int32_value()) { return constrainedVal; }
168  int32_t s = src.int32_value();
169 
170  // constrain if not within allowed range
171  if (s < min_) {
172  constrainedVal.set_int32_value(min_);
173  } else if (s > max_) {
174  constrainedVal.set_int32_value(max_);
175  } else if (!step_ || (s - min_) % step_ != 0) {
176  constrainedVal.set_int32_value(s - ((s - min_) % step_));
177  }
178  return constrainedVal;
