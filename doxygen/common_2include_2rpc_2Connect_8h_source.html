109 void updateResponse_(const std::string& oid, size_t idx, const IParam* p) override {
110 try {
111 // If Connect was cancelled, notify client and end process
112 if (this->isCancelled()) {
113 this->hasUpdate_ = true;
114 this->cv_.notify_one();
115 return;
116 }
117
118 if (this->authz_ != &catena::common::Authorizer::kAuthzDisabled && !this->authz_->readAuthz(*p)) {
119 return;
120 }
121
122 // Get all subscribed OIDs
123 auto subscribedOids = this->subscriptionManager_.getAllSubscribedOids(this->dm_);
124
125 // Check if we should process this update based on detail level
126 bool should_update = false;
127
128 // Map of detail levels to their update logic
129 const std::unordered_map<catena::Device_DetailLevel, std::function<void()>> detailLevelMap {
130 {catena::Device_DetailLevel_FULL, [&]() {
131 // Always update for FULL detail level
132 should_update = true;
133 }},
134 {catena::Device_DetailLevel_MINIMAL, [&]() {
135 // For MINIMAL, only update if it's in the minimal set
136 should_update = p->getDescriptor().minimalSet();
137 }},
138 {catena::Device_DetailLevel_SUBSCRIPTIONS, [&]() {
139 // Update if OID is subscribed or in minimal set
140 should_update = p->getDescriptor().minimalSet() ||
141 (std::find(subscribedOids.begin(), subscribedOids.end(), oid) != subscribedOids.end());
142 }},
143 {catena::Device_DetailLevel_COMMANDS, [&]() {
144 // For COMMANDS, only update command parameters
145 should_update = p->getDescriptor().isCommand();
146 }},
147 {catena::Device_DetailLevel_NONE, [&]() {
148 // Don't send any updates
149 should_update = false;
150 }}
151 };
152
153 if (detailLevelMap.contains(this->detailLevel_)) {
154 detailLevelMap.at(this->detailLevel_)();
155 } else {
156 should_update = false;
157 }
158
159 if (!should_update) {
160 return;
161 }
162
163
164 this->res_.mutable_value()->set_oid(oid);
165 this->res_.mutable_value()->set_element_index(idx);
166
167 catena::Value* value = this->res_.mutable_value()->mutable_value();
168
169 catena::exception_with_status rc{"", catena::StatusCode::OK};
170 rc = p->toProto(*value, *authz_);
171 //If the param conversion was successful, send the update
172 if (rc.status == catena::StatusCode::OK) {
173 this->hasUpdate_ = true;
174 this->cv_.notify_one();
175 }
176 } catch(catena::exception_with_status& why) {
177 // if an error is thrown, no update is pushed to the client
178 }
179 }

187 void updateResponse_(const ILanguagePack* l) override {
188 try {
189 // If Connect was cancelled, notify client and end process.
190 if (this->isCancelled()){
191 this->hasUpdate_ = true;
192 this->cv_.notify_one();
193 return;
194 }
195 // Returning if authorization is enabled and the client does not have monitor scope.
196 if (this->authz_ != &catena::common::Authorizer::kAuthzDisabled
197 && !this->authz_->hasAuthz(Scopes().getForwardMap().at(Scopes_e::kMonitor))) {
198 return;
199 }
200 // Updating res_'s device_component and pushing update.
201 auto pack = this->res_.mutable_device_component()->mutable_language_pack();
202 l->toProto(*pack->mutable_language_pack());
203 this->hasUpdate_ = true;
204 this->cv_.notify_one();
205 } catch(catena::exception_with_status& why){
206 // if an error is thrown, no update is pushed to the client
207 }
208 }
