133 void updateResponse_(const std::string& oid, const IParam* p, uint32_t slot) override {
134 try {
135 // If Connect was cancelled, notify client and end process
136 if (isCancelled()) {
137 hasUpdate_ = true;
138 cv_.notify_one();
139
140 } else if (authz_->readAuthz(*p)) {
141 // Map of detail levels to their update logic
142 const std::unordered_map<catena::Device_DetailLevel, std::function<bool()>> detailLevelMap {
143 {catena::Device_DetailLevel_FULL, [&]() {
144 // Always update for FULL detail level
145 return true;
146 }},
147 {catena::Device_DetailLevel_MINIMAL, [&]() {
148 // For MINIMAL, only update if it's in the minimal set
149 return p->getDescriptor().minimalSet();
150 }},
151 {catena::Device_DetailLevel_SUBSCRIPTIONS, [&]() {
152 // Update if OID is subscribed or in minimal set
153 return p->getDescriptor().minimalSet() || (dms_[slot] && subscriptionManager_.isSubscribed(oid, *dms_[slot]));
154 }},
155 {catena::Device_DetailLevel_COMMANDS, [&]() {
156 // For COMMANDS, only update command parameters
157 return p->getDescriptor().isCommand();
158 }},
159 {catena::Device_DetailLevel_NONE, [&]() {
160 // Don't send any updates
161 return false;
162 }}
163 };
164
165 if (detailLevelMap.contains(detailLevel_) && detailLevelMap.at(detailLevel_)()) {
166 res_.Clear();
167 res_.set_slot(slot);
168 res_.mutable_value()->set_oid(oid);
169 catena::Value* value = res_.mutable_value()->mutable_value();
170
171 catena::exception_with_status rc{"", catena::StatusCode::OK};
172 rc = p->toProto(*value, *authz_);
173 //If the param conversion was successful, send the update
174 if (rc.status == catena::StatusCode::OK) {
175 hasUpdate_ = true;
176 cv_.notify_one();
177 }
178 }
179 }
180 } catch(catena::exception_with_status& why) {
181 // if an error is thrown, no update is pushed to the client
182 DEBUG_LOG << "Failed to send SetValue update: " << why.what();
183 }
184 }

192 void updateResponse_(const ILanguagePack* l, uint32_t slot) override {
193 try {
194 // If Connect was cancelled, notify client and end process.
195 if (isCancelled()){
196 hasUpdate_ = true;
197 cv_.notify_one();
198 } else if (authz_->readAuthz(Scopes().getForwardMap().at(Scopes_e::kMonitor))) {
199 // Updating res_'s device_component and pushing update.
200 res_.Clear();
201 res_.set_slot(slot);
202 auto pack = res_.mutable_device_component()->mutable_language_pack();
203 l->toProto(*pack->mutable_language_pack());
204 hasUpdate_ = true;
205 cv_.notify_one();
206 }
207 } catch(catena::exception_with_status& why){
208 // if an error is thrown, no update is pushed to the client
209 DEBUG_LOG << "Failed to send language pack update: " << why.what();
210 }
211 }

222 void initAuthz_(const std::string& jwsToken, bool authz = false) override {
223 if (authz) {
224 sharedAuthz_ = std::make_shared<catena::common::Authorizer>(jwsToken);
225 authz_ = sharedAuthz_.get();
226 // Setting up their connection priority.
227 for (uint32_t i = static_cast<uint32_t>(Scopes_e::kAdmin); i >= static_cast<uint32_t>(Scopes_e::kMonitor); i -= 1) {
228 auto scope = Scopes().getForwardMap().at(static_cast<Scopes_e>(i));
229 // Read authz
230 if (authz_->hasAuthz(scope)) {
231 priority_ = 2 * i;
232 break;
233 // Write authz
234 } else if (authz_->hasAuthz(scope + ":w")) {
235 priority_ = 2 * i + 1;
236 if (forceConnection_ && static_cast<Scopes_e>(i) == Scopes_e::kAdmin) {
237 priority_ += 1; // forceConnection adds 1 to admin:w priority
238 }
239 break;
240 }
241 }
242 } else {
243 authz_ = &catena::common::Authorizer::kAuthzDisabled;
244 }
245 }
