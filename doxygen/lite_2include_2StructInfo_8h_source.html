213  auto fields = StructInfo<T>::fields; // get tuple of FieldInfo for this struct type
214  if (!src.has_struct_value()) {

220  // lambda function to call fromProto for the given field if it is authorized
221  auto writeField = [&](const auto& field) {
222  AuthzInfo subParamAuth = auth.subParamInfo(field.name);
