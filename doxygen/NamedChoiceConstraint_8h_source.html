83 NamedChoiceConstraint(ListInitializer init, bool strict, std::string oid, bool shared)

115 bool satisfied(const catena::Value& src) const override {
116 bool ans = false;
117 if constexpr (std::is_same<T, int32_t>::value) {
118 ans = choices_.contains(src.int32_value());
119 }
120 else if constexpr (std::is_same<T, std::string>::value) {
121 ans = !strict_ || choices_.contains(src.string_value());
122 }
123 return ans;
124 }

136 catena::Value apply(const catena::Value& src) const override {
137 return catena::Value();

144 void toProto(catena::Constraint& constraint) const override {
145 // Int choice serialization
146 if constexpr (std::is_same<T, int32_t>::value) {
147 constraint.set_type(catena::Constraint::INT_CHOICE);
148 for (auto& [value, name] : choices_) {
149 auto intChoice = constraint.mutable_int32_choice()->add_choices();
150 intChoice->set_value(value);
151 name.toProto(*intChoice->mutable_name());
152 }
153 }
154 // String choice serialization
155 else if constexpr (std::is_same<T, std::string>::value) {
156 // Default is STRING_CHOICE, changes to STRING_STRING_CHOICE if
157 // there are no display strings
158 constraint.set_type(catena::Constraint::STRING_CHOICE);
159 for (auto& [value, name] : choices_) {
160 auto stringChoice = constraint.mutable_string_string_choice()->add_choices();
161 stringChoice->set_value(value);
162 name.toProto(*stringChoice->mutable_name());
163 if (!name.displayStrings().empty()) {
164 constraint.set_type(catena::Constraint::STRING_STRING_CHOICE);
165 }
166 }
167 }
168 }
169
174 bool isRange() const override { return false; }
175
180 bool isShared() const override { return shared_; }
181
186 const std::string& getOid() const override { return oid_; }

188 private:

191 T default_;
192 bool shared_;
193 std::string oid_;
