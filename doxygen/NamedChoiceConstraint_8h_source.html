126 bool satisfied(const catena::Value& src) const override {
127 // INT_CHOICE implementation
128 if constexpr(isIntChoice_) {
129 return choices_.contains(src.int32_value());
130 }
131 // STRING_CHOICE and STRING_STRING_CHOICE implementation
132 else if constexpr(isStringChoice_ || isStringStringChoice_) {
133 return !strict_ || choices_.contains(src.string_value());
134 }
135 // Implementation for everything else.
136 else {
137 return false;
138 }

159 void toProto(catena::Constraint& constraint) const override {
160 // INT_CHOICE implementation
161 if constexpr(isIntChoice_) {
162 constraint.set_type(catena::Constraint::INT_CHOICE);
163 for (auto& [value, name] : choices_) {
164 auto intChoice = constraint.mutable_int32_choice()->add_choices();
165 intChoice->set_value(value);
166 name.toProto(*intChoice->mutable_name());
167 }
168 }
169 // STRING_CHOICE implementation
170 else if constexpr(isStringChoice_) {
171 constraint.set_type(catena::Constraint::STRING_CHOICE);
172 for (auto& [value, name] : choices_) {
173 *constraint.mutable_string_choice()->add_choices() = value;
174 }
175 }
176 //STRING_STRING_CHOICE implementation
177 else if constexpr(isStringStringChoice_) {
178 constraint.set_type(catena::Constraint::STRING_STRING_CHOICE);
179 for (auto& [value, name] : choices_) {
180 auto stringChoice = constraint.mutable_string_string_choice()->add_choices();
181 stringChoice->set_value(value);
182 name.toProto(*stringChoice->mutable_name());
183 }
184 }
185 }
