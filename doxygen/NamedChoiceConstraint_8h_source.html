116 bool satisfied(const catena::Value& src) const override {

118 if constexpr (std::is_same<T, int32_t>::value) {
119 return choices_.contains(src.int32_value());

121
122 if constexpr (std::is_same<T, std::string>::value) {
123 if (!strict_) {
124 return true;

149 void toProto(catena::Constraint& constraint) const override {
150
151 if constexpr (std::is_same<T, int32_t>::value) {
152 constraint.set_type(catena::Constraint::INT_CHOICE);
153 for (auto& [value, name] : choices_) {
154 auto intChoice = constraint.mutable_int32_choice()->add_choices();
155 intChoice->set_value(value);
156 name.toProto(*intChoice->mutable_name());
157 }
158 }
159
160 if constexpr (std::is_same<T, std::string>::value) {
161 constraint.set_type(catena::Constraint::STRING_STRING_CHOICE);
162 for (auto& [value, name] : choices_) {
163 auto stringChoice = constraint.mutable_string_string_choice()->add_choices();
164 stringChoice->set_value(value);
165 name.toProto(*stringChoice->mutable_name());
166 }
167 }
168 }
