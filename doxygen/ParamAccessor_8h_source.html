228  template <bool Threadsafe> std::unique_ptr<ParamAccessor> subParam(const std::string& fieldName) {
229  using LockGuard = std::conditional_t<Threadsafe, std::lock_guard<Mutex>, common::FakeLock>;
230  LockGuard lock(deviceModel_.get().mutex_);
231  Param& parent = param_.get();
232  if (!parent.mutable_params()->contains(fieldName)) {
233  BAD_STATUS("subParam called on non-existent field", catena::StatusCode::INVALID_ARGUMENT);
234  }
235  Param& childParam = parent.mutable_params()->at(fieldName);
236  Value& value = value_.get();
237  // If child doesn't have a scope defined, use the parent's scope
238  const std::string& scope = childParam.access_scope() == "" ? this->scope_ : childParam.access_scope();
239 
240  DeviceModel::ParamAccessorData pad;
241  Value* v;
242  if (value.kind_case() == Value::KindCase::kStructValue) {
243  // field is a struct
244  if (!value.struct_value().fields().contains(fieldName)) {
245  BAD_STATUS("subParam called on non-existent field", catena::StatusCode::INVALID_ARGUMENT);
246  }
247  v = value.mutable_struct_value()->mutable_fields()->at(fieldName).mutable_value();
248  } else if (value.kind_case() == Value::KindCase::kStructVariantValue) {
249  // field is a variant
250  v = value.mutable_struct_variant_value()->mutable_value();
251  } else {
252  // field is a simple or simple array type
253  BAD_STATUS("subParam called on non-struct or variant type", catena::StatusCode::INVALID_ARGUMENT);
254  }
255  std::get<0>(pad) = &childParam;
256  std::get<1>(pad) = v ? v : &DeviceModel::noValue_;
257 
258  return std::unique_ptr<ParamAccessor>(new ParamAccessor{deviceModel_.get(), pad, oid_, scope});
259  }

320  template <bool Threadsafe = true, typename V> void getValue(V& dst) const {
321  try {
322  using LockGuard = std::conditional_t<Threadsafe, std::lock_guard<Mutex>, catena::common::FakeLock>;
323  LockGuard lock(deviceModel_.get().mutex_);
324  auto& getter = Getter::getInstance();
325  if constexpr (catena::full::has_getStructInfo<V>) {
326  // dst is a struct
327  const auto& structInfo = dst.getStructInfo();
328  char* base = reinterpret_cast<char*>(&dst);
329  auto* srcFields = value_.get().mutable_struct_value()->mutable_fields();
330  for (auto& field : structInfo.fields) {
331  char* dstAddr = base + field.offset;
332  const Value& srcField = srcFields->at(field.name).value();
333  if (srcFields->contains(field.name)) {
334  const catena::Value::KindCase kc = srcField.kind_case();
335  if (kc == Value::KindCase::kStructValue) {
336  // field is a struct
337  const std::unique_ptr<ParamAccessor> sp = subParam<false>(field.name);
338  field.wrapGetter(dstAddr, sp.get());
339  } else {
340  // field is a simple or simple array type
341  getter[kc](dstAddr, &srcField);
342  }
343  }
344  }
345  } else if constexpr (catena::meta::is_variant<V>::value) {
346  // dst is a variant gather information about the protobuf source
347  const StructVariantValue& src = value_.get().struct_variant_value();
348  const std::string& variant = src.struct_variant_type();
349  Value::KindCase kc = src.value().kind_case();
350 
351  // gather info about the native destination
352  auto& variantInfoFunctory = catena::full::ParamAccessor::VariantInfoGetter::getInstance();
353  const catena::full::VariantInfo& variantInfo = variantInfoFunctory[std::type_index(typeid(V))]();
354  const catena::full::VariantMemberInfo vmi = variantInfo.members.at(variant);
355 
356  // set the variant to the correct type, and return a pointer to it
357  void* ptr = vmi.set(&dst);
358  if (kc == Value::KindCase::kStructValue) {
359  // variant value is a struct
360  const std::unique_ptr<ParamAccessor> sp = subParam<false>(variant);
361  vmi.wrapGetter(ptr, sp.get());
362  } else if (kc == Value::KindCase::kStructVariantValue) {
363  // variant value is a variant
364  BAD_STATUS("Variant of variant not supported", catena::StatusCode::INVALID_ARGUMENT);
365  } else {
366  // field is a simple or simple array type
367  getter[kc](reinterpret_cast<char*>(ptr), &src.value());
368  }
369  } else {
370  // dst is a simple type or whole array
371  getter[getKindCase<V>(dst)](&dst, &value_.get());
372  }
373  } catch (const catena::exception_with_status& why) {
374  std::stringstream err;
375  err << "getValue failed: " << why.what() << '\n' << __PRETTY_FUNCTION__ << '\n';
376  throw catena::exception_with_status(err.str(), why.status);
377  } catch (const std::runtime_error& why) {

434  template <bool Threadsafe = true, typename V> void setValue(const V& src) {
435  try {
436  using LockGuard = std::conditional_t<Threadsafe, std::lock_guard<Mutex>, catena::common::FakeLock>;
437  LockGuard lock(deviceModel_.get().mutex_);
438  auto& setter = Setter::getInstance();
439  if constexpr (catena::full::has_getStructInfo<V>) {
440  const auto& structInfo = src.getStructInfo();
441  const char* base = reinterpret_cast<const char*>(&src);
442  auto* dstFields = value_.get().mutable_struct_value()->mutable_fields();
443  for (auto& field : structInfo.fields) {
444  const char* srcAddr = base + field.offset;
445  if (!dstFields->contains(field.name)) {
446  dstFields->insert({field.name, StructField{}});
447  *dstFields->at(field.name).mutable_value() = Value{};
448  std::unique_ptr<ParamAccessor> sp = subParam<false>(field.name);
449  field.wrapSetter(sp.get(), srcAddr);
450  } else {
451  Value* dstField = dstFields->at(field.name).mutable_value();
452  if (dstField->kind_case() == Value::KindCase::kStructValue) {
453  // field is a struct
454  std::unique_ptr<ParamAccessor> sp = subParam<false>(field.name);
455  field.wrapSetter(sp.get(), srcAddr);
456  } else {
457  // field is a simple or simple array type
458  setter[dstField->kind_case()](dstField, srcAddr);
459  }
460  }
461  }
462  } else if constexpr (catena::meta::is_variant<V>::value) {
463  auto& variantInfoFunctory = ParamAccessor::VariantInfoGetter::getInstance();
464  const catena::full::VariantInfo& variantInfo = variantInfoFunctory[std::type_index(typeid(V))]();
465  Value& v = value_.get();
466  StructVariantValue* vv = v.mutable_struct_variant_value();
467  std::string* currentVariant = vv->mutable_struct_variant_type();
468  const std::string& variant = variantInfo.lookup[src.index()];
469  const std::unique_ptr<ParamAccessor> sp = subParam<false>(variant);
470  if (variant.compare(*currentVariant) != 0) {
471  // we need to change the variant type in the protobuf
472  *currentVariant = variant;
473  }
474  variantInfo.members.at(variant).wrapSetter(sp.get(), &src);
475  } else {
476  setter[getKindCase<V>(src)](&value_.get(), &src);
477  }
478  deviceModel_.get().pushUpdates.emit(*this, kParamEnd);
479  } catch (const catena::exception_with_status& why) {
