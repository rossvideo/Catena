213  template <bool Threadsafe> std::unique_ptr<ParamAccessor> subParam(const std::string& fieldName) {
214  using LockGuard = std::conditional_t<Threadsafe, std::lock_guard<Mutex>, FakeLock>;
215  LockGuard lock(deviceModel_.get().mutex_);
216  Param& parent = param_.get();
217  Param& childParam = parent.mutable_params()->at(fieldName);
218  Value& value = value_.get();
219  DeviceModel::ParamAccessorData pad{};
220  if (value.kind_case() == Value::KindCase::kStructValue) {
221  // field is a struct
222  if (!value.struct_value().fields().contains(fieldName)) {
223  BAD_STATUS("subParam called on non-existent field", catena::StatusCode::INVALID_ARGUMENT);
224  }
225  Value* v = value.mutable_struct_value()->mutable_fields()->at(fieldName).mutable_value();
226  pad = {&childParam, v};
227  } else if (value.kind_case() == Value::KindCase::kStructVariantValue) {
228  // field is a variant
229  Value* v = value.mutable_struct_variant_value()->mutable_value();
230  pad = {&childParam, v};
231  } else {
232  // field is a simple or simple array type
233  BAD_STATUS("subParam called on non-struct or variant type", catena::StatusCode::INVALID_ARGUMENT);
234  }
235  return std::unique_ptr<ParamAccessor>(new ParamAccessor{deviceModel_.get(), pad, oid_ + "/" + fieldName});
236  }

291  template <bool Threadsafe = true, typename V> void setValue(const V& src) {
292  try {
293  using LockGuard = std::conditional_t<Threadsafe, std::lock_guard<Mutex>, catena::FakeLock>;
294  LockGuard lock(deviceModel_.get().mutex_);
295  auto& setter = Setter::getInstance();
296  if constexpr (catena::has_getStructInfo<V>) {
297  const auto& structInfo = src.getStructInfo();
298  const char* base = reinterpret_cast<const char*>(&src);
299  auto* dstFields = value_.get().mutable_struct_value()->mutable_fields();
300  for (auto& field : structInfo.fields) {
301  const char* srcAddr = base + field.offset;
302  if (!dstFields->contains(field.name)) {
303  dstFields->insert({field.name, StructField{}});
304  *dstFields->at(field.name).mutable_value() = Value{};
305  std::unique_ptr<ParamAccessor> sp = subParam<false>(field.name);
306  field.wrapSetter(sp.get(), srcAddr);
307  } else {
308  Value* dstField = dstFields->at(field.name).mutable_value();
309  if (dstField->kind_case() == Value::KindCase::kStructValue) {
310  // field is a struct
311  std::unique_ptr<ParamAccessor> sp = subParam<false>(field.name);
312  field.wrapSetter(sp.get(), srcAddr);
313  } else {
314  // field is a simple or simple array type
315 
316  setter[dstField->kind_case()](dstField, srcAddr);
317  }
318  }
319  }
320  } else if constexpr (catena::meta::is_variant<V>::value) {
321  auto& variantInfoFunctory = catena::ParamAccessor::VariantInfoGetter::getInstance();
322  const catena::VariantInfo& variantInfo = variantInfoFunctory[std::type_index(typeid(V))]();
323  Value& v = value_.get();
324  StructVariantValue* vv = v.mutable_struct_variant_value();
325  std::string* currentVariant = vv->mutable_struct_variant_type();
326  const std::string& variant = variantInfo.lookup[src.index()];
327  const std::unique_ptr<ParamAccessor> sp = subParam<false>(variant);
328  if (variant.compare(*currentVariant) != 0) {
329  // we need to change the variant type in the protobuf
330  *currentVariant = variant;
331  }
332  variantInfo.members.at(variant).wrapSetter(sp.get(), &src);
333  } else {
334  typename std::remove_const<typename std::remove_reference<decltype(src)>::type>::type x;
335  setter[getKindCase(x)](&value_.get(), &src);
336  }
337  deviceModel_.get().valueSetByService(*this, kParamEnd);
338  } catch (const std::runtime_error& why) {
339  // most likely thrown by the setter encountering a missing function
340  std::stringstream err;
341  err << "setValue failed: " << why.what() << '\n' << __PRETTY_FUNCTION__ << '\n';
342  throw catena::exception_with_status(err.str(), catena::StatusCode::UNIMPLEMENTED);
343  } catch (const catena::exception_with_status& why) {
344  std::stringstream err;
345  err << "setValue failed: " << why.what() << '\n' << __PRETTY_FUNCTION__ << '\n';
346  throw catena::exception_with_status(err.str(), why.status);
347  } catch (...) {
348  std::stringstream err;
349  err << "setValue failed with unknown exception: " << '\n' << __PRETTY_FUNCTION__ << '\n';
350  throw catena::exception_with_status(err.str(), catena::StatusCode::UNKNOWN);
351  }
352  }

403  template <bool Threadsafe = true, typename V> void getValue(V& dst) const {
404  try {
405  using LockGuard = std::conditional_t<Threadsafe, std::lock_guard<Mutex>, catena::FakeLock>;
406  LockGuard lock(deviceModel_.get().mutex_);
407  auto& getter = Getter::getInstance();
408  if constexpr (catena::has_getStructInfo<V>) {
409  // dst is a struct
410  const auto& structInfo = dst.getStructInfo();
411  char* base = reinterpret_cast<char*>(&dst);
412  auto* srcFields = value_.get().mutable_struct_value()->mutable_fields();
413  for (auto& field : structInfo.fields) {
414  char* dstAddr = base + field.offset;
415  const Value& srcField = srcFields->at(field.name).value();
416  if (srcFields->contains(field.name)) {
417  const catena::Value::KindCase kc = srcField.kind_case();
418  if (kc == Value::KindCase::kStructValue) {
419  // field is a struct
420  const std::unique_ptr<ParamAccessor> sp = subParam<false>(field.name);
421  field.wrapGetter(dstAddr, sp.get());
422  } else {
423  // field is a simple or simple array type
424  getter[kc](dstAddr, &srcField);
425  }
426  }
427  }
428  } else if constexpr (catena::meta::is_variant<V>::value) {
429  // dst is a variant gather information about the protobuf source
430  const StructVariantValue& src = value_.get().struct_variant_value();
431  const std::string& variant = src.struct_variant_type();
432  Value::KindCase kc = src.value().kind_case();
433 
434  // gather info about the native destination
435  auto& variantInfoFunctory = catena::ParamAccessor::VariantInfoGetter::getInstance();
436  const catena::VariantInfo& variantInfo = variantInfoFunctory[std::type_index(typeid(V))]();
437  const catena::VariantMemberInfo vmi = variantInfo.members.at(variant);
438 
439  // set the variant to the correct type, and return a pointer to it
440  void* ptr = vmi.set(&dst);
441  if (kc == Value::KindCase::kStructValue) {
442  // variant value is a struct
443  const std::unique_ptr<ParamAccessor> sp = subParam<false>(variant);
444  vmi.wrapGetter(ptr, sp.get());
445  } else if (kc == Value::KindCase::kStructVariantValue) {
446  // variant value is a variant
447  BAD_STATUS("Variant of variant not supported", catena::StatusCode::INVALID_ARGUMENT);
448  } else {
449  // field is a simple or simple array type
450  getter[kc](reinterpret_cast<char*>(ptr), &src.value());
451  }
452  } else {
453  // dst is a simple type
454  getter[getKindCase<V>(dst)](&dst, &value_.get());
455  }
456  } catch (const std::runtime_error& why) {
457  // most likely thrown by the getter encountering a missing function
458  std::stringstream err;
459  err << "setValue failed: " << why.what() << '\n' << __PRETTY_FUNCTION__ << '\n';
460  throw catena::exception_with_status(err.str(), catena::StatusCode::UNIMPLEMENTED);
461  } catch (const catena::exception_with_status& why) {
462  std::stringstream err;
463  err << "setValue failed: " << why.what() << '\n' << __PRETTY_FUNCTION__ << '\n';
464  throw catena::exception_with_status(err.str(), why.status);
465  } catch (...) {
466  std::stringstream err;
467  err << "setValue failed with unknown exception: " << '\n' << __PRETTY_FUNCTION__ << '\n';
468  throw catena::exception_with_status(err.str(), catena::StatusCode::UNKNOWN);
469  }
