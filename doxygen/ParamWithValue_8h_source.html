490 template<meta::IsVector U>
491 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
492 using ElemType = U::value_type;
493 if (!oid.front_is_index()) {
494 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
495 return nullptr;
496 }
497 size_t oidIndex = oid.front_as_index();
498 oid.pop();
499
500 if (oidIndex >= value.size() || oidIndex == catena::common::Path::kEnd) {
501 // If index is out of bounds, return nullptr
502 status = catena::exception_with_status("Index out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
503 return nullptr;
504 }
505
506 if (oid.empty()) {
507 // we reached the end of the path, return the element
508 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
509 } else {
510 if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
511 // The path has more segments, keep recursing
512 return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
513 } else {
514 // This type is not a CatenaStructArray so it has no sub-params
515 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
516 return nullptr;
517 }
518 }

531 template <CatenaStruct U>
532 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
533 if (!oid.front_is_string()) {
534 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
535 return nullptr;
536 }
537 std::string oidStr = oid.front_as_string();
538 oid.pop();
539 auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
540
541 std::unique_ptr<IParam> ip = findParamByName_<U>(fields, oidStr);
542 if (!ip) {
543 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
544 return nullptr;
545 }
546
547 if (oid.empty()) {
548 // we reached the end of the path, return the element
549 return ip;
550 } else {
551 // The path has more segments, keep recursing
552 return ip->getParam(oid, authz, status);
553 }

617 template <meta::IsVariant U>
618 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
619 if (!oid.front_is_string()) {
620 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
621 return nullptr;
622 }
623 std::string oidStr = oid.front_as_string();
624 oid.pop();
625 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
626 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
627 return nullptr;
628 }
629
630 if (oid.empty()) {
631 // we reached the end of the path, return the element
632 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
633 using V = std::decay_t<decltype(arg)>;
634 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
635 }, value);
636 } else {
637 // The path has more segments, keep recursing
638 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
639 using V = std::decay_t<decltype(arg)>;
640 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
641 }, value);
642 }

674 std::size_t str_length(const U& value) const { return 0; }

676 // Tracker initializers. One overload for each case.
682 template<typename U>
683 catena::exception_with_status initializeTracker_(const U& value) {
684 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
685 try {
686 // 0 unless value is a string.
687 *mSizeTracker_ = str_length(value);
688 } catch (...) {
689 ans = catena::exception_with_status("Could not initialize tracker for " + descriptor_.getOid(), catena::StatusCode::INTERNAL);
690 }
691 return ans;
692 }
700 template<meta::IsVector U>
701 catena::exception_with_status initializeTracker_(const U& arrayValue) {
702 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
703 try {
704 *mSizeTracker_ = arrayValue.size();
705 // STRING_ARRAY only functionality.
706 if (descriptor_.type() == catena::ParamType::STRING_ARRAY) {
707 // Initializing/clearing the tSizeTracker.
708 if (!tSizeTracker_) {
709 tSizeTracker_ = std::make_shared<TSizeTracker>();
710 } else {
711 tSizeTracker_->clear();
712 }
713 // Setting mSizeTracker and tSizeTracker.
714 for (auto& value : arrayValue) {
715 tSizeTracker_->push_back(str_length(value));
716 }
717 }
718 } catch (...) {
719 ans = catena::exception_with_status("Could not initialize tracker for " + descriptor_.getOid(), catena::StatusCode::INTERNAL);
720 }
721 return ans;
722 }

724 // Tracker updaters. One overload for each case.
734 template<typename U, typename V>
735 catena::exception_with_status updateTracker_(U& oldValue, const V& newValue, Path::Index index) {
736 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
737 // Index cannot be defined.
738 if (index != Path::kNone) {
739 ans = catena::exception_with_status("Index not applicable to setValue for param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
740 // newValue must be the same type at oldValue.
741 } else if (!std::is_same<U, V>::value) {
742 ans = catena::exception_with_status("Type mismatch between setValue and param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
743 // Calling initializeTracker_ if valid input.
744 } else {
745 ans = initializeTracker_(newValue);
746 }
747 return ans;
748 }
759 template<typename U, typename V> requires (!meta::IsVector<V>)
760 catena::exception_with_status updateTracker_(std::vector<U>& arrayValue, const V& value, Path::Index index) {
761 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
762 // Index must be specified.
763 if (index == Path::kNone) {
764 ans = catena::exception_with_status("Index not specified in SetValue call to " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
765 // Value must be the same type as what's stored in the array.
766 } else if (!std::is_same<U, V>::value) {
767 ans = catena::exception_with_status("Type mismatch between setValue and param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
768 // Index must be within bounds.
769 } else if (index >= *mSizeTracker_ && index != Path::kEnd) {
770 ans = catena::exception_with_status("Index out of bounds of array " + descriptor_.getOid(), catena::StatusCode::OUT_OF_RANGE);
771 // Case 1: Appending to the end.
772 } else if (index == Path::kEnd) {
773 *mSizeTracker_ += 1;
774 if (descriptor_.type() == catena::ParamType::STRING_ARRAY) {
775 tSizeTracker_->push_back(str_length(value));
776 }
777 // Case 2: Inserting (do nothing unless string array).
778 } else if (descriptor_.type() == catena::ParamType::STRING_ARRAY) {
779 tSizeTracker_->at(index) = str_length(value);
780 }
781 return ans;
782 }
783 using Kind = catena::Value::KindCase;

796 }},
797 {Kind::kFloat32Value, [this](const catena::Value& value, Path::Index index) {

799 }},
800 {Kind::kStringValue, [this](const catena::Value& value, Path::Index index) {

802 }},
803 {Kind::kStructValue, [this](const catena::Value& value, Path::Index index) {

833 public: // Validation functions.
841 bool validateSetValue(const catena::Value& value, Path::Index index, Authorizer& authz, catena::exception_with_status& ans) override {
842 // Checking read permissions.
843 if (!authz.readAuthz(*this)) {
844 ans = catena::exception_with_status("Not authorized to read the param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
845 }
846 // Checking write permissions.
847 else if (!authz.writeAuthz(*this)) {
848 ans = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
849 // Currently: If unimplemented ignore.
850 } else if (updateTrackerMap_.contains(value.kind_case())) {
851 // Updating trackers.
852 ans = updateTrackerMap_.at(value.kind_case())(value, index);
853 // If the above was successful, check mSize and tSize.
854 if (ans.status == catena::StatusCode::OK) {
855 // Validating max length.
856 if (*mSizeTracker_ > descriptor_.max_length()) {
857 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
858 // Validating total length if applicable.
859 } else if (tSizeTracker_) {
860 std::size_t totalLength = 0;
861 for (std::size_t length : *tSizeTracker_) {
862 totalLength += length;
863 }
864 if (totalLength > descriptor_.total_length()) {
865 ans = catena::exception_with_status("String array param " + descriptor_.getOid() + " exceeds total length", catena::StatusCode::OUT_OF_RANGE);
866 }
867 }
868 }
869 }
870 return ans.status == catena::StatusCode::OK;
871 }
875 void resetValidate() override {
876 // Resetting the trackers.
877 std::shared_ptr<std::size_t> mSizeTracker_ = std::make_shared<std::size_t>(0);
878 std::shared_ptr<TSizeTracker> tSizeTracker_ = nullptr;
879 // Reinitializing the trackers with the param's actual value.
880 initializeTracker_(value_.get());
881 }
