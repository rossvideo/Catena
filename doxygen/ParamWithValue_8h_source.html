386 template<meta::IsVector U>
387 std::unique_ptr<IParam> addBack(U& value, Authorizer& authz, catena::exception_with_status& status) {
388 using ElemType = U::value_type;
389 auto oidIndex = value.size();
390 if (!authz.writeAuthz(*this)) {
391 status = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
392 return nullptr;
393 } else if (oidIndex >= descriptor_.max_length()) {
394 status = catena::exception_with_status("Array " + descriptor_.getOid() + " at maximum capacity", catena::StatusCode::OUT_OF_RANGE);
395 return nullptr;
396 } else {
397 value.push_back(ElemType());
398 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
399 }

468 template<meta::IsVector U>
469 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
470 using ElemType = U::value_type;
471 if (!oid.front_is_index()) {
472 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
473 return nullptr;
474 }
475 size_t oidIndex = oid.front_as_index();
476 oid.pop();
477
478 if (oidIndex >= value.size() || oidIndex == catena::common::Path::kEnd) {
479 // If index is out of bounds, return nullptr
480 status = catena::exception_with_status("Index " + std::to_string(oidIndex) + " out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
481 return nullptr;
482 }
483
484 if (oid.empty()) {
485 // we reached the end of the path, return the element
486 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
487 } else {
488 if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
489 // The path has more segments, keep recursing
490 return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
491 } else {
492 // This type is not a CatenaStructArray so it has no sub-params
493 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
494 return nullptr;
495 }
496 }

509 template <CatenaStruct U>
510 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
511 if (!oid.front_is_string()) {
512 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
513 return nullptr;
514 }
515 std::string oidStr = oid.front_as_string();
516 oid.pop();
517 auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
518
519 std::unique_ptr<IParam> ip = findParamByName_<U>(fields, oidStr);
520 if (!ip) {
521 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
522 return nullptr;
523 }
524
525 if (oid.empty()) {
526 // we reached the end of the path, return the element
527 return ip;
528 } else {
529 // The path has more segments, keep recursing
530 return ip->getParam(oid, authz, status);
531 }

595 template <meta::IsVariant U>
596 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
597 if (!oid.front_is_string()) {
598 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
599 return nullptr;
600 }
601 std::string oidStr = oid.front_as_string();
602 oid.pop();
603 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
604 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
605 return nullptr;
606 }
607
608 if (oid.empty()) {
609 // we reached the end of the path, return the element
610 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
611 using V = std::decay_t<decltype(arg)>;
612 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
613 }, value);
614 } else {
615 // The path has more segments, keep recursing
616 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
617 using V = std::decay_t<decltype(arg)>;
618 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
619 }, value);
620 }
621 }

654 // Tracker updaters. One overload for each case.
665 template<typename U, typename V>
666 catena::exception_with_status validateSetValue_(U& oldVal, const V& newVal, const catena::Value& protoVal, Path::Index index, Authorizer& authz) {
667 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
668 // Index cannot be defined.
669 if (index != Path::kNone) {
670 ans = catena::exception_with_status("Index not applicable to setValue for param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
671 // newVal must pass all validation checks in validFromProto.
672 } else if (validFromProto(protoVal, &oldVal, descriptor_, ans, authz)) {
673 // Valid case if validFromProto returns true.
674 mSizeTracker_ = std::make_shared<std::size_t>(str_length(newVal));
675 }
676
677 // If above was sucessful, finish by validating mSize
678 if (ans.status == catena::StatusCode::OK && *mSizeTracker_ > descriptor_.max_length()) {
679 // String cannot exceed max_length().
680 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
681 }
682
683 return ans;
684 }
695 template<typename U, typename V> requires (!meta::IsVector<V>)
696 catena::exception_with_status validateSetValue_(std::vector<U>& arrayVal, const V& memVal, const catena::Value& protoVal, Path::Index index, Authorizer& authz) {
697 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
698 // Initializing mSizeTracker_ and tSizeTracker if they're not already.
699 if (!mSizeTracker_) {
700 mSizeTracker_ = std::make_shared<std::size_t>(arrayVal.size());
701 tSizeTracker_ = std::make_shared<TSizeTracker>();
702 for (auto& memVal : arrayVal) {
703 tSizeTracker_->push_back(str_length(memVal));
704 }
705 }
706
707 U testVal; // testVal for validFromProto() routing.
708 // Index must be specified.
709 if (index == Path::kNone) {
710 ans = catena::exception_with_status("Index not specified in SetValue call to " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
711 // Index must be within bounds.
712 } else if (index != Path::kEnd && index >= *mSizeTracker_) {
713 ans = catena::exception_with_status("Index " + std::to_string(index) + " out of bounds of array " + descriptor_.getOid(), catena::StatusCode::OUT_OF_RANGE);
714 // memVal must pass all validation checks in validFromProto.
715 } else if (validFromProto(protoVal, &testVal, descriptor_, ans, authz)) {
716 // Valid case if validFromProto returns true.
717 // Valid case 1: Appending to the end.
718 if (index == Path::kEnd) {
719 *mSizeTracker_ += 1;
720 tSizeTracker_->push_back(str_length(memVal));
721 // Valid case 2: Inserting at an existing index.
722 } else {
723 tSizeTracker_->at(index) = str_length(memVal);
724 }
725 }
726
727 // If above was sucessful, finish by validating mSize and tSize
728 if (ans.status == catena::StatusCode::OK) {
729 // Array cannot exceed max_length().
730 if (*mSizeTracker_ > descriptor_.max_length()) {
731 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
732 // Sum of strings in array cannot exceed total_length().
733 } else {
734 std::size_t totalLength = 0;
735 for (std::size_t length : *tSizeTracker_) {
736 totalLength += length;
737 }
738 if (totalLength > descriptor_.total_length()) {
739 ans = catena::exception_with_status("String array param " + descriptor_.getOid() + " exceeds total length", catena::StatusCode::OUT_OF_RANGE);
740 }
741 }
742 }
743
744 return ans;
745 }
