381 template<meta::IsVector U>
382 std::unique_ptr<IParam> addBack(U& value, Authorizer& authz, catena::exception_with_status& status) {
383 using ElemType = U::value_type;
384 auto oidIndex = value.size();
385 if (!authz.writeAuthz(*this)) {
386 status = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
387 return nullptr;
388 } else if (oidIndex >= descriptor_.max_length()) {
389 status = catena::exception_with_status("Array " + descriptor_.getOid() + " at maximum capacity", catena::StatusCode::OUT_OF_RANGE);

391 } else {
392 value.push_back(ElemType());
393 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);

463 template<meta::IsVector U>
464 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
465 using ElemType = U::value_type;
466 if (!oid.front_is_index()) {
467 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
468 return nullptr;
469 }
470 size_t oidIndex = oid.front_as_index();
471 oid.pop();
472
473 if (oidIndex >= value.size() || oidIndex == catena::common::Path::kEnd) {
474 // If index is out of bounds, return nullptr
475 status = catena::exception_with_status("Index " + std::to_string(oidIndex) + " out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
476 return nullptr;
477 }
478
479 if (oid.empty()) {
480 // we reached the end of the path, return the element
481 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
482 } else {
483 if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
484 // The path has more segments, keep recursing
485 return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
486 } else {
487 // This type is not a CatenaStructArray so it has no sub-params
488 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
489 return nullptr;
490 }
491 }

504 template <CatenaStruct U>
505 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
506 if (!oid.front_is_string()) {
507 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
508 return nullptr;
509 }
510 std::string oidStr = oid.front_as_string();
511 oid.pop();
512 auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
513
514 std::unique_ptr<IParam> ip = findParamByName_<U>(fields, oidStr);
515 if (!ip) {
516 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
517 return nullptr;
518 }
519
520 if (oid.empty()) {
521 // we reached the end of the path, return the element
522 return ip;
523 } else {
524 // The path has more segments, keep recursing
525 return ip->getParam(oid, authz, status);
526 }

590 template <meta::IsVariant U>
591 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
592 if (!oid.front_is_string()) {
593 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
594 return nullptr;
595 }
596 std::string oidStr = oid.front_as_string();
597 oid.pop();
598 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
599 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
600 return nullptr;
601 }
602
603 if (oid.empty()) {
604 // we reached the end of the path, return the element
605 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
606 using V = std::decay_t<decltype(arg)>;
607 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
608 }, value);
609 } else {
610 // The path has more segments, keep recursing
611 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
612 using V = std::decay_t<decltype(arg)>;
613 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
614 }, value);
615 }

660 template<typename U, typename V>
661 catena::exception_with_status validateSetValue_(U& oldVal, const V& newVal, const catena::Value& protoVal, Path::Index index, Authorizer& authz) {
662 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
663 // Index cannot be defined.
664 if (index != Path::kNone) {
665 ans = catena::exception_with_status("Index not applicable to setValue for param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
666 // newVal must pass all validation checks in validFromProto.
667 } else if (validFromProto(protoVal, &oldVal, descriptor_, ans, authz)) {
668 // Valid case if validFromProto returns true.
669 mSizeTracker_ = std::make_shared<std::size_t>(str_length(newVal));
670 }
671
672 // If above was sucessful, finish by validating mSize
673 if (ans.status == catena::StatusCode::OK && *mSizeTracker_ > descriptor_.max_length()) {
674 // String cannot exceed max_length().
675 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
676 }
677
678 return ans;
679 }
690 template<typename U, typename V> requires (!meta::IsVector<V>)
691 catena::exception_with_status validateSetValue_(std::vector<U>& arrayVal, const V& memVal, const catena::Value& protoVal, Path::Index index, Authorizer& authz) {
692 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
693 // Initializing mSizeTracker_ and tSizeTracker if they're not already.
694 if (!mSizeTracker_) {
695 mSizeTracker_ = std::make_shared<std::size_t>(arrayVal.size());
696 tSizeTracker_ = std::make_shared<TSizeTracker>();
697 for (auto& memVal : arrayVal) {
698 tSizeTracker_->push_back(str_length(memVal));
699 }
700 }
701
702 U testVal; // testVal for validFromProto() routing.
703 // Index must be specified.
704 if (index == Path::kNone) {
705 ans = catena::exception_with_status("Index not specified in SetValue call to " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
706 // Index must be within bounds.
707 } else if (index != Path::kEnd && index >= *mSizeTracker_) {
708 ans = catena::exception_with_status("Index " + std::to_string(index) + " out of bounds of array " + descriptor_.getOid(), catena::StatusCode::OUT_OF_RANGE);
709 // memVal must pass all validation checks in validFromProto.
710 } else if (validFromProto(protoVal, &testVal, descriptor_, ans, authz)) {
711 // Valid case if validFromProto returns true.
712 // Valid case 1: Appending to the end.
713 if (index == Path::kEnd) {
714 *mSizeTracker_ += 1;
715 tSizeTracker_->push_back(str_length(memVal));
716 // Valid case 2: Inserting at an existing index.
717 } else {
718 tSizeTracker_->at(index) = str_length(memVal);
719 }
720 }
721
722 // If above was sucessful, finish by validating mSize and tSize
723 if (ans.status == catena::StatusCode::OK) {
724 // Array cannot exceed max_length().
725 if (*mSizeTracker_ > descriptor_.max_length()) {
726 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
727 // Sum of strings in array cannot exceed total_length().
728 } else {
729 std::size_t totalLength = 0;
730 for (std::size_t length : *tSizeTracker_) {
731 totalLength += length;
732 }
733 if (totalLength > descriptor_.total_length()) {
734 ans = catena::exception_with_status("String array param " + descriptor_.getOid() + " exceeds total length", catena::StatusCode::OUT_OF_RANGE);
735 }
736 }
737 }
738
739 return ans;
740 }
