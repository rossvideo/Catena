288  template<meta::IsVector U>
289  std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
290  using ElemType = U::value_type;
291  if (!oid.front_is_index()) {
292  status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
293  return nullptr;
294  }
295  size_t oidIndex = oid.front_as_index();
296  oid.pop();
297 
298  if (oidIndex == catena::common::Path::kEnd) {
299  // Index is "-", add a new element to the array
300  oidIndex = value.size();
301  value.push_back(ElemType());
302  } else if (oidIndex >= value.size()) {
303  // If index is out of bounds, return nullptr
304  status = catena::exception_with_status("Index out of bounds in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
305  return nullptr;
306  }
307 
308  if (oid.empty()) {
309  // we reached the end of the path, return the element
310  return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
311  } else {
312  if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
313  // The path has more segments, keep recursing
314  return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
315  } else {
316  // This type is not a CatenaStructArray so it has no sub-params
317  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
318  return nullptr;
319  }
320  }
321  }

333  template <CatenaStruct U>
334  std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
335  if (!oid.front_is_string()) {
336  status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
337  return nullptr;
338  }
339  std::string oidStr = oid.front_as_string();
340  oid.pop();
341  auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
342 
343  std::unique_ptr<IParam> ip = findParamByName_<U>(fields, oidStr);
344  if (!ip) {
345  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
346  return nullptr;
347  }
348 
349  if (oid.empty()) {
350  // we reached the end of the path, return the element
351  return ip;
352  } else {
353  // The path has more segments, keep recursing
354  return ip->getParam(oid, authz, status);

386  template <typename Struct, typename Tuple, std::size_t... Is>
387  std::unique_ptr<IParam> findParamByNameImpl_(const Tuple& fields, const std::string& name, std::index_sequence<Is...>) {
393  std::unique_ptr<IParam> params[] = {std::get<Is>(fields).name == name ? getParamAtIndex_<Is>(fields) : nullptr ...};
394  for (auto& param : params) {

419  template <meta::IsVariant U>
420  std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
421  if (!oid.front_is_string()) {
422  status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
423  return nullptr;
424  }
425  std::string oidStr = oid.front_as_string();
426  oid.pop();
427 
428  if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
429  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
430  return nullptr;
431  }
432 
433  if (oid.empty()) {
434  // we reached the end of the path, return the element
435  return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
436  using V = std::decay_t<decltype(arg)>;
437  return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
438  }, value);
439  } else {
440  // The path has more segments, keep recursing
441  return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
442  using V = std::decay_t<decltype(arg)>;
443  return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
444  }, value);
445  }
446  }
