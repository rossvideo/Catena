220 catena::exception_with_status ans{"", catena::StatusCode::OK};
221 if (!authz.readAuthz(*this)) {
222 ans = catena::exception_with_status("Param does not exist ", catena::StatusCode::NOT_FOUND);
223 }
224 else if (!authz.writeAuthz(*this)) {
225 ans = catena::exception_with_status("Not authorized to write to param ", catena::StatusCode::PERMISSION_DENIED);
226 }
227 else if (!validateSize(value)) {
228 ans = catena::exception_with_status("Value exceeds maximum length ", catena::StatusCode::OUT_OF_RANGE);
229 } else {
230 catena::common::fromProto<T>(value, &value_.get(), descriptor_, authz);
231 }

399 template<meta::IsVector U>
400 std::unique_ptr<IParam> addBack(U& value, Authorizer& authz, catena::exception_with_status& status) {
401 using ElemType = U::value_type;
402 auto oidIndex = value.size();
403 if (!authz.writeAuthz(*this)) {
404 status = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
405 return nullptr;
406 } else if (oidIndex >= descriptor_.max_length()) {
407 status = catena::exception_with_status("Array " + descriptor_.getOid() + " at maximum capacity ", catena::StatusCode::OUT_OF_RANGE);
408 return nullptr;
409 } else {
410 value.push_back(ElemType());
411 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
412 }
413 }

481 template<meta::IsVector U>
482 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
483 using ElemType = U::value_type;
484 if (!oid.front_is_index()) {
485 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
486 return nullptr;
487 }
488 size_t oidIndex = oid.front_as_index();
489 oid.pop();
490
491 if (oidIndex >= value.size() || oidIndex == catena::common::Path::kEnd) {
492 // If index is out of bounds, return nullptr
493 status = catena::exception_with_status("Index out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
494 return nullptr;
495 }
496
497 if (oid.empty()) {
498 // we reached the end of the path, return the element
499 return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
500 } else {
501 if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
502 // The path has more segments, keep recursing
503 return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
504 } else {
505 // This type is not a CatenaStructArray so it has no sub-params
506 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist ", catena::StatusCode::NOT_FOUND);
507 return nullptr;
508 }
509 }
510 }

522 template <CatenaStruct U>
523 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
524 if (!oid.front_is_string()) {
525 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
526 return nullptr;
527 }
528 std::string oidStr = oid.front_as_string();
529 oid.pop();
530 auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
531
532 std::unique_ptr<IParam> ip = findParamByName_<U>(fields, oidStr);
533 if (!ip) {
534 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
535 return nullptr;
536 }
537
538 if (oid.empty()) {
539 // we reached the end of the path, return the element
540 return ip;
541 } else {
542 // The path has more segments, keep recursing
543 return ip->getParam(oid, authz, status);
544 }
545 }

559 template <typename Struct, typename Tuple>
560 std::unique_ptr<IParam> findParamByName_(const Tuple& fields, const std::string& name) {
561 return findParamByNameImpl_<Struct>(fields, name, std::make_index_sequence<std::tuple_size_v<Tuple>>());
562 }

575 template <typename Struct, typename Tuple, std::size_t... Is>
576 std::unique_ptr<IParam> findParamByNameImpl_(const Tuple& fields, const std::string& name, std::index_sequence<Is...>) {
582 std::unique_ptr<IParam> params[] = {std::get<Is>(fields).name == name ? getParamAtIndex_<Is>(fields) : nullptr ...};
583 for (auto& param : params) {
584 if (param) {
585 // returns first IParam unique_ptr element that isn't a nullptr
586 return std::move(param);
587 }
588 }
589 return nullptr;
590 }

601 template <std::size_t I, typename Tuple>
602 std::unique_ptr<IParam> getParamAtIndex_(const Tuple& tuple) {
603 // Get the type of the field at index I so that we can create a ParamWithValue object of that type
604 using FieldType = typename std::tuple_element_t<I, Tuple>::Field;
605 return std::make_unique<ParamWithValue<FieldType>>(std::get<I>(tuple), value_.get(), descriptor_);

608 template <meta::IsVariant U>
609 std::unique_ptr<IParam> getParam_(Path& oid, U& value, Authorizer& authz, catena::exception_with_status& status) {
610 if (!oid.front_is_string()) {
611 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
612 return nullptr;
613 }
614 std::string oidStr = oid.front_as_string();
615 oid.pop();
616 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
617 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist ", catena::StatusCode::NOT_FOUND);
618 return nullptr;
619 }
620
621 if (oid.empty()) {
622 // we reached the end of the path, return the element
623 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
624 using V = std::decay_t<decltype(arg)>;
625 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
626 }, value);
627 } else {
628 // The path has more segments, keep recursing
629 return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
630 using V = std::decay_t<decltype(arg)>;
631 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
632 }, value);
633 }
634 }
