416 template<meta::IsVector U>
417 std::unique_ptr<IParam> addBack_(U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
418 using ElemType = U::value_type;
419 std::unique_ptr<IParam> returnParam = nullptr;
420 auto oidIndex = value.size();
421 // Check writeAuthz.
422 if (!authz.writeAuthz(*this)) {
423 status = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
424 // Make sure add does not exceed max length constraint.
425 } else if (oidIndex >= descriptor_.max_length()) {
426 status = catena::exception_with_status("Array " + descriptor_.getOid() + " at maximum capacity", catena::StatusCode::OUT_OF_RANGE);
427 } else {
428 value.push_back(ElemType());
429 returnParam = std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
430 }

500 template<meta::IsVector U>
501 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
502 using ElemType = U::value_type;
503 std::unique_ptr<IParam> returnParam = nullptr;
504 // Make sure the front is an index
505 if (!oid.front_is_index()) {
506 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
507 } else {
508 size_t oidIndex = oid.front_as_index();
509 oid.pop();
510 // If index is out of bounds, return nullptr
511 if (oidIndex >= value.size()) {
512 status = catena::exception_with_status("Index " + std::to_string(oidIndex) + " out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
513 // we reached the end of the path, return the element
514 } else if (oid.empty()) {
515 returnParam = std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
516 // The path has more segments, keep recursing
517 } else if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
518 // The path has more segments, keep recursing
519 returnParam = ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
520 // This type is not a CatenaStructArray so it has no sub-params
521 } else {
522 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
523 }
524 }
525 return returnParam;
526 }

538 template <CatenaStruct U>
539 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
540 std::unique_ptr<IParam> returnParam = nullptr;
541 // Make sure the front is a field name.
542 if (!oid.front_is_string()) {
543 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
544 } else {
545 std::string oidStr = oid.front_as_string();
546 oid.pop();
547 auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
548 returnParam = findParamByName_<U>(fields, oidStr);
549 // Param does not exist, return nullptr
550 if (!returnParam) {
551 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
552 // The path has more segments, keep recursing
553 } else if (!oid.empty()) {
554 returnParam = returnParam->getParam(oid, authz, status);
555 // If we're at the end of path make sure we have readAuthz for the gotten param before returning.
556 } else if (!authz.readAuthz(*returnParam)) {
557 status = catena::exception_with_status("Not authorized to read param " + oid.fqoid(), catena::StatusCode::PERMISSION_DENIED);
558 returnParam = nullptr;
559 }
560 }
561 return returnParam;
562 }

574 template <meta::IsVariant U>
575 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
576 std::unique_ptr<IParam> returnParam = nullptr;
577 if (!oid.front_is_string()) {
578 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
579 } else {
580 std::string oidStr = oid.front_as_string();
581 oid.pop();
582 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
583 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
584 // we reached the end of the path, return the element
585 } else if (oid.empty()) {
586 returnParam = std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
587 using V = std::decay_t<decltype(arg)>;
588 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
589 }, value);
590 // Making sure we have readAuthz for the gotten param before returning.
591 if (!authz.readAuthz(*returnParam)) {
592 status = catena::exception_with_status("Not authorized to read param " + oid.fqoid(), catena::StatusCode::PERMISSION_DENIED);
593 returnParam = nullptr;
594 }
595 // The path has more segments, keep recursing
596 } else {
597 returnParam = std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
598 using V = std::decay_t<decltype(arg)>;
599 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
600 }, value);
601 }
602 }
603 return returnParam;
604 }

678 template<typename U, typename V>
679 catena::exception_with_status validateSetValue_(U& oldVal, const V& newVal, const st2138::Value& protoVal, Path::Index index, const IAuthorizer& authz) {
680 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
681 // Index cannot be defined.
682 if (index != Path::kNone) {
683 ans = catena::exception_with_status("Index not applicable to setValue for param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
684 // newVal must pass all validation checks in validFromProto.
685 } else if (validFromProto(protoVal, &oldVal, descriptor_, ans, authz)) {
686 // Valid case if validFromProto returns true.
687 mSizeTracker_ = std::make_shared<std::size_t>(size_(newVal));
688 }
689
690 // If above was sucessful, finish by validating mSize
691 if (ans.status == catena::StatusCode::OK && *mSizeTracker_ > descriptor_.max_length()) {
692 // String cannot exceed max_length().
693 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
694 }
695
696 return ans;
697 }
708 template<typename U, typename V> requires (!meta::IsVector<V>)
709 catena::exception_with_status validateSetValue_(std::vector<U>& arrayVal, const V& memVal, const st2138::Value& protoVal, Path::Index index, const IAuthorizer& authz) {
710 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
711 // Initializing mSizeTracker_ and tSizeTracker if they're not already.
712 if (!mSizeTracker_) {
713 mSizeTracker_ = std::make_shared<std::size_t>(arrayVal.size());
714 }
715 if (!tSizeTracker_) {
716 tSizeTracker_ = std::make_shared<TSizeTracker>();
717 for (auto& memVal : arrayVal) {
718 tSizeTracker_->push_back(size_(memVal));
719 }
720 }
721
722 U testVal; // testVal for validFromProto() routing.
723 // Index must be specified.
724 if (index == Path::kNone) {
725 ans = catena::exception_with_status("Index not specified in SetValue call to " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
726 // Index must be within bounds.
727 } else if (index != Path::kEnd && index >= *mSizeTracker_) {
728 ans = catena::exception_with_status("Index " + std::to_string(index) + " out of bounds of array " + descriptor_.getOid(), catena::StatusCode::OUT_OF_RANGE);
729 // memVal must pass all validation checks in validFromProto.
730 } else if (validFromProto(protoVal, &testVal, descriptor_, ans, authz)) {
731 // Valid case if validFromProto returns true.
732 // Valid case 1: Appending to the end.
733 if (index == Path::kEnd) {
734 *mSizeTracker_ += 1;
735 tSizeTracker_->push_back(size_(memVal));
736 // Valid case 2: Inserting at an existing index.
737 } else {
738 tSizeTracker_->at(index) = size_(memVal);
739 }
740 }
741
742 // If above was sucessful, finish by validating mSize and tSize
743 if (ans.status == catena::StatusCode::OK) {
744 // Array cannot exceed max_length().
745 if (*mSizeTracker_ > descriptor_.max_length()) {
746 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
747 // Sum of strings in array cannot exceed total_length().
748 } else {
749 std::size_t totalLength = 0;
750 for (std::size_t length : *tSizeTracker_) {
751 totalLength += length;
752 }
753 if (totalLength > descriptor_.total_length()) {
754 ans = catena::exception_with_status("String array param " + descriptor_.getOid() + " exceeds total length", catena::StatusCode::OUT_OF_RANGE);
755 }
756 }
757 }
758
759 return ans;
760 }
