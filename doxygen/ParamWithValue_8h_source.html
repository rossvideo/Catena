433 template<meta::IsVector U>
434 std::unique_ptr<IParam> addBack_(U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
435 using ElemType = U::value_type;
436 std::unique_ptr<IParam> returnParam = nullptr;
437 auto oidIndex = value.size();
438 // Check writeAuthz.
439 if (!authz.writeAuthz(*this)) {
440 status = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
441 // Make sure add does not exceed max length constraint.
442 } else if (oidIndex >= descriptor_.max_length()) {
443 status = catena::exception_with_status("Array " + descriptor_.getOid() + " at maximum capacity", catena::StatusCode::OUT_OF_RANGE);
444 } else {
445 value.push_back(ElemType());
446 returnParam = std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
447 }
448 return returnParam;

523 template<meta::IsVector U>
524 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
525 using ElemType = U::value_type;
526 std::unique_ptr<IParam> returnParam = nullptr;
527 // Make sure the front is an index
528 if (!oid.front_is_index()) {
529 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
530 } else {
531 size_t oidIndex = oid.front_as_index();
532 oid.pop();
533 // If index is out of bounds, return nullptr
534 if (oidIndex >= value.size()) {
535 status = catena::exception_with_status("Index " + std::to_string(oidIndex) + " out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
536 // we reached the end of the path, return the element
537 } else if (oid.empty()) {
538 returnParam = std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
539 // The path has more segments, keep recursing
540 } else if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
541 // The path has more segments, keep recursing
542 returnParam = ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
543 // This type is not a CatenaStructArray so it has no sub-params
544 } else {
545 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
546 }
547 }
548 return returnParam;
549 }

564 template <CatenaStruct U>
565 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
566 std::unique_ptr<IParam> returnParam = nullptr;
567 // Make sure the front is a field name.
568 if (!oid.front_is_string()) {
569 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
570 } else {
571 std::string oidStr = oid.front_as_string();
572 oid.pop();
573 auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
574 returnParam = findParamByName_<U>(fields, oidStr);
575 // Param does not exist, return nullptr
576 if (!returnParam) {
577 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
578 // The path has more segments, keep recursing
579 } else if (!oid.empty()) {
580 returnParam = returnParam->getParam(oid, authz, status);
581 // If we're at the end of path make sure we have readAuthz for the gotten param before returning.
582 } else if (!authz.readAuthz(*returnParam)) {
583 status = catena::exception_with_status("Not authorized to read param " + oid.fqoid(), catena::StatusCode::PERMISSION_DENIED);
584 returnParam = nullptr;
585 }
586 }
587 return returnParam;

603 template <meta::IsVariant U>
604 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
605 std::unique_ptr<IParam> returnParam = nullptr;
606 if (!oid.front_is_string()) {
607 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
608 } else {
609 std::string oidStr = oid.front_as_string();
610 oid.pop();
611 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
612 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
613 // we reached the end of the path, return the element
614 } else if (oid.empty()) {
615 returnParam = std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
616 using V = std::decay_t<decltype(arg)>;
617 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
618 }, value);
619 // Making sure we have readAuthz for the gotten param before returning.
620 if (!authz.readAuthz(*returnParam)) {
621 status = catena::exception_with_status("Not authorized to read param " + oid.fqoid(), catena::StatusCode::PERMISSION_DENIED);
622 returnParam = nullptr;
623 }
624 // The path has more segments, keep recursing
625 } else {
626 returnParam = std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
627 using V = std::decay_t<decltype(arg)>;
628 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
629 }, value);
630 }
631 }
632 return returnParam;
633 }

707 template<typename U, typename V>
708 catena::exception_with_status validateSetValue_(U& oldVal, const V& newVal, const catena::Value& protoVal, Path::Index index, const IAuthorizer& authz) {
709 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
710 // Index cannot be defined.
711 if (index != Path::kNone) {
712 ans = catena::exception_with_status("Index not applicable to setValue for param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
713 // newVal must pass all validation checks in validFromProto.
714 } else if (validFromProto(protoVal, &oldVal, descriptor_, ans, authz)) {
715 // Valid case if validFromProto returns true.
716 mSizeTracker_ = std::make_shared<std::size_t>(size_(newVal));
717 }
718
719 // If above was sucessful, finish by validating mSize
720 if (ans.status == catena::StatusCode::OK && *mSizeTracker_ > descriptor_.max_length()) {
721 // String cannot exceed max_length().
722 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
723 }
724
725 return ans;
726 }
737 template<typename U, typename V> requires (!meta::IsVector<V>)
738 catena::exception_with_status validateSetValue_(std::vector<U>& arrayVal, const V& memVal, const catena::Value& protoVal, Path::Index index, const IAuthorizer& authz) {
739 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
740 // Initializing mSizeTracker_ and tSizeTracker if they're not already.
741 if (!mSizeTracker_) {
742 mSizeTracker_ = std::make_shared<std::size_t>(arrayVal.size());
743 }
744 if (!tSizeTracker_) {
745 tSizeTracker_ = std::make_shared<TSizeTracker>();
746 for (auto& memVal : arrayVal) {
747 tSizeTracker_->push_back(size_(memVal));
748 }
749 }
750
751 U testVal; // testVal for validFromProto() routing.
752 // Index must be specified.
753 if (index == Path::kNone) {
754 ans = catena::exception_with_status("Index not specified in SetValue call to " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
755 // Index must be within bounds.
756 } else if (index != Path::kEnd && index >= *mSizeTracker_) {
757 ans = catena::exception_with_status("Index " + std::to_string(index) + " out of bounds of array " + descriptor_.getOid(), catena::StatusCode::OUT_OF_RANGE);
758 // memVal must pass all validation checks in validFromProto.
759 } else if (validFromProto(protoVal, &testVal, descriptor_, ans, authz)) {
760 // Valid case if validFromProto returns true.
761 // Valid case 1: Appending to the end.
762 if (index == Path::kEnd) {
763 *mSizeTracker_ += 1;
764 tSizeTracker_->push_back(size_(memVal));
765 // Valid case 2: Inserting at an existing index.
766 } else {
767 tSizeTracker_->at(index) = size_(memVal);
768 }
769 }
770
771 // If above was sucessful, finish by validating mSize and tSize
772 if (ans.status == catena::StatusCode::OK) {
773 // Array cannot exceed max_length().
774 if (*mSizeTracker_ > descriptor_.max_length()) {
775 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
776 // Sum of strings in array cannot exceed total_length().
777 } else {
778 std::size_t totalLength = 0;
779 for (std::size_t length : *tSizeTracker_) {
780 totalLength += length;
781 }
782 if (totalLength > descriptor_.total_length()) {
783 ans = catena::exception_with_status("String array param " + descriptor_.getOid() + " exceeds total length", catena::StatusCode::OUT_OF_RANGE);
784 }
785 }
786 }
787
788 return ans;
789 }
