436 template<meta::IsVector U>
437 std::unique_ptr<IParam> addBack_(U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
438 using ElemType = U::value_type;
439 std::unique_ptr<IParam> returnParam = nullptr;
440 auto oidIndex = value.size();
441 // Check writeAuthz.
442 if (!authz.writeAuthz(*this)) {
443 status = catena::exception_with_status("Not authorized to write to param " + descriptor_.getOid(), catena::StatusCode::PERMISSION_DENIED);
444 // Make sure add does not exceed max length constraint.
445 } else if (oidIndex >= descriptor_.max_length()) {
446 status = catena::exception_with_status("Array " + descriptor_.getOid() + " at maximum capacity", catena::StatusCode::OUT_OF_RANGE);
447 } else {
448 value.push_back(ElemType());
449 returnParam = std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
450 }
451 return returnParam;

521 std::unique_ptr<IParam> getParam_(Path& oid, EmptyValue& value, const IAuthorizer& authz, catena::exception_with_status& status) {
522 std::unique_ptr<IParam> returnParam = nullptr;
523 // Make sure the front is a field name.
524 if (!oid.front_is_string()) {
525 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
526 } else {
527 std::string oidStr = oid.front_as_string();
528 oid.pop();
529
530 // Check if this parameter has sub-parameters defined in its descriptor
531 const auto& subParams = descriptor_.getAllSubParams();
532 auto it = subParams.find(oidStr);
533 if (it != subParams.end()) {
534 // Found sub-parameter in descriptor, create a ParamWithValue for it
535 returnParam = std::make_unique<ParamWithValue<EmptyValue>>(value, *it->second);
536 // The path has more segments, keep recursing
537 if (!oid.empty()) {
538 returnParam = returnParam->getParam(oid, authz, status);
539 // If we're at the end of path make sure we have readAuthz for the gotten param before returning.
540 } else if (!authz.readAuthz(*returnParam)) {
541 status = catena::exception_with_status("Not authorized to read param " + oid.fqoid(), catena::StatusCode::PERMISSION_DENIED);
542 returnParam = nullptr;
543 }
544 } else {
545 // Param does not exist, return nullptr
546 status = catena::exception_with_status("Param " + oidStr + " does not exist", catena::StatusCode::NOT_FOUND);
547 }
548 }
549 return returnParam;
550 }

565 template<meta::IsVector U>
566 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
567 using ElemType = U::value_type;
568 std::unique_ptr<IParam> returnParam = nullptr;
569 // Make sure the front is an index
570 if (!oid.front_is_index()) {
571 status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
572 } else {
573 size_t oidIndex = oid.front_as_index();
574 oid.pop();
575 // If index is out of bounds, return nullptr
576 if (oidIndex >= value.size()) {
577 status = catena::exception_with_status("Index " + std::to_string(oidIndex) + " out of bounds in path " + oid.fqoid(), catena::StatusCode::OUT_OF_RANGE);
578 // we reached the end of the path, return the element
579 } else if (oid.empty()) {
580 returnParam = std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
581 // The path has more segments, keep recursing
582 } else if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
583 // The path has more segments, keep recursing
584 returnParam = ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, authz, status);
585 // This type is not a CatenaStructArray so it has no sub-params
586 } else {
587 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
588 }
589 }
590 return returnParam;
591 }

606 template <CatenaStruct U>
607 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
608 std::unique_ptr<IParam> returnParam = nullptr;
609 // Make sure the front is a field name.
610 if (!oid.front_is_string()) {

620 // The path has more segments, keep recursing
621 } else if (!oid.empty()) {
622 returnParam = returnParam->getParam(oid, authz, status);
623 // If we're at the end of path make sure we have readAuthz for the gotten param before returning.

645 template <meta::IsVariant U>
646 std::unique_ptr<IParam> getParam_(Path& oid, U& value, const IAuthorizer& authz, catena::exception_with_status& status) {
647 std::unique_ptr<IParam> returnParam = nullptr;
648 if (!oid.front_is_string()) {
649 status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
650 } else {
651 std::string oidStr = oid.front_as_string();
652 oid.pop();
653 if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
654 status = catena::exception_with_status("Param " + oid.fqoid() + " does not exist", catena::StatusCode::NOT_FOUND);
655 // we reached the end of the path, return the element
656 } else if (oid.empty()) {
657 returnParam = std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
658 using V = std::decay_t<decltype(arg)>;
659 return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
660 }, value);
661 // Making sure we have readAuthz for the gotten param before returning.
662 if (!authz.readAuthz(*returnParam)) {
663 status = catena::exception_with_status("Not authorized to read param " + oid.fqoid(), catena::StatusCode::PERMISSION_DENIED);
664 returnParam = nullptr;
665 }
666 // The path has more segments, keep recursing
667 } else {
668 returnParam = std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
669 using V = std::decay_t<decltype(arg)>;
670 return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, authz, status);
671 }, value);
672 }
673 }
674 return returnParam;
675 }

749 template<typename U, typename V>
750 catena::exception_with_status validateSetValue_(U& oldVal, const V& newVal, const st2138::Value& protoVal, Path::Index index, const IAuthorizer& authz) {
751 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
752 // Index cannot be defined.
753 if (index != Path::kNone) {
754 ans = catena::exception_with_status("Index not applicable to setValue for param " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
755 // newVal must pass all validation checks in validFromProto.
756 } else if (validFromProto(protoVal, &oldVal, descriptor_, ans, authz)) {
757 // Valid case if validFromProto returns true.
758 mSizeTracker_ = std::make_shared<std::size_t>(size_(newVal));
759 }
760
761 // If above was sucessful, finish by validating mSize
762 if (ans.status == catena::StatusCode::OK && *mSizeTracker_ > descriptor_.max_length()) {
763 // String cannot exceed max_length().
764 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
765 }
766
767 return ans;
768 }
779 template<typename U, typename V> requires (!meta::IsVector<V>)
780 catena::exception_with_status validateSetValue_(std::vector<U>& arrayVal, const V& memVal, const st2138::Value& protoVal, Path::Index index, const IAuthorizer& authz) {
781 catena::exception_with_status ans{"OK", catena::StatusCode::OK};
782 // Initializing mSizeTracker_ and tSizeTracker if they're not already.
783 if (!mSizeTracker_) {
784 mSizeTracker_ = std::make_shared<std::size_t>(arrayVal.size());
785 }
786 if (!tSizeTracker_) {
787 tSizeTracker_ = std::make_shared<TSizeTracker>();
788 for (auto& memVal : arrayVal) {
789 tSizeTracker_->push_back(size_(memVal));
790 }
791 }
792
793 U testVal; // testVal for validFromProto() routing.
794 // Index must be specified.
795 if (index == Path::kNone) {
796 ans = catena::exception_with_status("Index not specified in SetValue call to " + descriptor_.getOid(), catena::StatusCode::INVALID_ARGUMENT);
797 // Index must be within bounds.
798 } else if (index != Path::kEnd && index >= *mSizeTracker_) {
799 ans = catena::exception_with_status("Index " + std::to_string(index) + " out of bounds of array " + descriptor_.getOid(), catena::StatusCode::OUT_OF_RANGE);
800 // memVal must pass all validation checks in validFromProto.
801 } else if (validFromProto(protoVal, &testVal, descriptor_, ans, authz)) {
802 // Valid case if validFromProto returns true.
803 // Valid case 1: Appending to the end.
804 if (index == Path::kEnd) {
805 *mSizeTracker_ += 1;
806 tSizeTracker_->push_back(size_(memVal));
807 // Valid case 2: Inserting at an existing index.
808 } else {
809 tSizeTracker_->at(index) = size_(memVal);
810 }
811 }
812
813 // If above was sucessful, finish by validating mSize and tSize
814 if (ans.status == catena::StatusCode::OK) {
815 // Array cannot exceed max_length().
816 if (*mSizeTracker_ > descriptor_.max_length()) {
817 ans = catena::exception_with_status("Param " + descriptor_.getOid() + " exceeds maximum capacity", catena::StatusCode::OUT_OF_RANGE);
818 // Sum of strings in array cannot exceed total_length().
819 } else {
820 std::size_t totalLength = 0;
821 for (std::size_t length : *tSizeTracker_) {
822 totalLength += length;
823 }
824 if (totalLength > descriptor_.total_length()) {
825 ans = catena::exception_with_status("String array param " + descriptor_.getOid() + " exceeds total length", catena::StatusCode::OUT_OF_RANGE);
826 }
827 }
828 }
829
830 return ans;
