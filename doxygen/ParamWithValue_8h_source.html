291  template<meta::IsVector U>
292  std::unique_ptr<IParam> getParam_(Path& oid, U& value, catena::exception_with_status& status) {
293  using ElemType = U::value_type;
294  if (!oid.front_is_index()) {
295  status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
296  return nullptr;
297  }
298  size_t oidIndex = oid.front_as_index();
299  oid.pop();
300 
301  if (oidIndex == catena::common::Path::kEnd) {
302  // Index is "-", add a new element to the array
303  oidIndex = value.size();
304  value.push_back(ElemType());
305  } else if (oidIndex >= value.size()) {
306  // If index is out of bounds, return nullptr
307  status = catena::exception_with_status("Index out of bounds in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
308  return nullptr;
309  }
310 
311  if (oid.empty()) {
312  // we reached the end of the path, return the element
313  return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
314  } else {
315  if constexpr (CatenaStruct<ElemType> || meta::IsVariant<ElemType>) {
316  // The path has more segments, keep recursing
317  return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, status);
318  } else {
319  // This type is not a CatenaStructArray so it has no sub-params
320  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
321  return nullptr;
322  }
323  }

336  template <CatenaStruct U>
337  std::unique_ptr<IParam> getParam_(Path& oid, U& value, catena::exception_with_status& status) {
338  if (!oid.front_is_string()) {
339  status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
340  return nullptr;
341  }
342  std::string oidStr = oid.front_as_string();
343  oid.pop();
344  auto fields = StructInfo<U>::fields; // get tuple of FieldInfo for this struct type
345 
346  std::unique_ptr<IParam> ip = findParamByName_<U>(fields, oidStr);
347  if (!ip) {
348  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
349  return nullptr;
350  }
351 
352  if (oid.empty()) {
353  // we reached the end of the path, return the element
354  return ip;
355  } else {
356  // The path has more segments, keep recursing
357  return ip->getParam(oid, status);
358  }

389  template <typename Struct, typename Tuple, std::size_t... Is>
390  std::unique_ptr<IParam> findParamByNameImpl_(const Tuple& fields, const std::string& name, std::index_sequence<Is...>) {
396  std::unique_ptr<IParam> params[] = {std::get<Is>(fields).name == name ? getParamAtIndex_<Is>(fields) : nullptr ...};
397  for (auto& param : params) {
398  if (param) {
399  // returns first IParam unique_ptr element that isn't a nullptr
400  return std::move(param);
401  }
402  }
403  return nullptr;

422  template <meta::IsVariant U>
423  std::unique_ptr<IParam> getParam_(Path& oid, U& value, catena::exception_with_status& status) {
424  if (!oid.front_is_string()) {
425  status = catena::exception_with_status("Expected string in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
426  return nullptr;
427  }
428  std::string oidStr = oid.front_as_string();
429  oid.pop();
430 
431  if (catena::common::alternativeNames<U>[value.index()] != oidStr) {
432  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
433  return nullptr;
434  }
435 
436  if (oid.empty()) {
437  // we reached the end of the path, return the element
438  return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
439  using V = std::decay_t<decltype(arg)>;
440  return std::make_unique<ParamWithValue<V>>(arg, descriptor_.getSubParam(oidStr));
441  }, value);
442  } else {
443  // The path has more segments, keep recursing
444  return std::visit([&](auto& arg) -> std::unique_ptr<IParam> {
445  using V = std::decay_t<decltype(arg)>;
446  return ParamWithValue<V>(arg, descriptor_.getSubParam(oidStr)).getParam(oid, status);
447  }, value);
448  }
449  }
