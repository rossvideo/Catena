280  template<meta::IsVector U>
281  std::unique_ptr<IParam> getParam_(Path& oid, U& value, catena::exception_with_status& status) {
282  using ElemType = U::value_type;
283  if (!oid.front_is_index()) {
284  status = catena::exception_with_status("Expected index in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
285  return nullptr;
286  }
287  size_t oidIndex = oid.front_as_index();
288  oid.pop();
289 
290  if (oidIndex == catena::common::Path::kEnd) {
291  // Index is "-", add a new element to the array
292  oidIndex = value.size();
293  value.push_back(ElemType());
294  } else if (oidIndex >= value.size()) {
295  // If index is out of bounds, return nullptr
296  status = catena::exception_with_status("Index out of bounds in path " + oid.fqoid(), catena::StatusCode::INVALID_ARGUMENT);
297  return nullptr;
298  }
299 
300  if (oid.empty()) {
301  // we reached the end of the path, return the element
302  return std::make_unique<ParamWithValue<ElemType>>(value[oidIndex], descriptor_);
303  } else {
304  if constexpr (CatenaStruct<ElemType>) {
305  // The path has more segments, keep recursing
306  return ParamWithValue<ElemType>(value[oidIndex], descriptor_).getParam(oid, status);
307  } else {
308  // This type is not a CatenaStructArray so it has no sub-params
309  status = catena::exception_with_status("Param does not exist", catena::StatusCode::INVALID_ARGUMENT);
310  return nullptr;
311  }
