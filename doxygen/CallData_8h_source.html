106 std::string jwsToken_() const override {
107 std::string jwsToken = "";
108 if (service_->authorizationEnabled()) {
109 // Getting client metadata from context.
110 auto clientMeta = &context_.client_metadata();
111 if (clientMeta != nullptr) {
112 // Getting authorization data (JWS bearer token) from client metadata.
113 auto authData = clientMeta->find("authorization");
114 if (authData != clientMeta->end() && authData->second.starts_with("Bearer ")) {
115 // Getting token (after "bearer") and returning as an std::string.
116 auto tokenSubStr = authData->second.substr(std::string("Bearer ").length());
117 jwsToken = std::string(tokenSubStr.begin(), tokenSubStr.end());
118 } else {
119 throw catena::exception_with_status("JWS bearer token not found", catena::StatusCode::UNAUTHENTICATED);
120 }
121 } else {
122 throw catena::exception_with_status("Client metadata not found", catena::StatusCode::UNAUTHENTICATED);
123 }
124 }
125 return jwsToken;
126 }

131 void processTimestamps_() {
132 // Getting request receival time formatted as,
133 // <number of milliseconds since start of epoch>
134 const auto epoch_time = std::chrono::system_clock::now().time_since_epoch();
135 requestReceived_ = std::chrono::duration_cast<std::chrono::milliseconds>(epoch_time).count();

137 auto clientMeta = context_.client_metadata();
138 // Getting client metadata from context.

140 if (kv != clientMeta.end()) {
141 std::string value(kv->second.data(), kv->second.size());
142 catena::readTimestamp(value, requestStart_);
143 }
