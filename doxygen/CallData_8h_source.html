84 std::string jwsToken_() const override {
85 std::string jwsToken = "";
86 if (service_->authorizationEnabled()) {
87 // Getting client metadata from context.
88 auto clientMeta = &context_.client_metadata();
89 if (clientMeta != nullptr) {
90 // Getting authorization data (JWS bearer token) from client metadata.
91 auto authData = clientMeta->find("authorization");
92 if (authData != clientMeta->end() && authData->second.starts_with("Bearer ")) {
93 // Getting token (after "bearer") and returning as an std::string.
94 auto tokenSubStr = authData->second.substr(std::string("Bearer ").length());
95 jwsToken = std::string(tokenSubStr.begin(), tokenSubStr.end());
96 } else {
97 throw catena::exception_with_status("JWS bearer token not found", catena::StatusCode::UNAUTHENTICATED);
98 }
99 } else {
100 throw catena::exception_with_status("Client metadata not found", catena::StatusCode::UNAUTHENTICATED);
101 }
102 }
103 return jwsToken;
104 }
