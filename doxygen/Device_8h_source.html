350 template <typename TAG>
351 void addItem(const std::string& key, typename TAG::type* item) {
352 if constexpr (std::is_same_v<TAG, common::ParamTag>) {
353 params_[key] = item;
354 }
355 if constexpr (std::is_same_v<TAG, common::ConstraintTag>) {
356 constraints_[key] = item;
357 }
358 if constexpr (std::is_same_v<TAG, common::MenuGroupTag>) {
359 menu_groups_[key] = item;
360 }
361 if constexpr (std::is_same_v<TAG, common::CommandTag>) {
362 commands_[key] = item;
363 }
364 if constexpr (std::is_same_v<TAG, common::LanguagePackTag>) {
365 language_packs_[key] = item;
366 }
367 }
