336 inline void unhandled_exception() {
337 exception_ = std::current_exception();

342 inline void rethrow_if_exception() {

348 };

350 CommandResponder(std::coroutine_handle<promise_type> h) : handle_(h) {}
351 CommandResponder(const CommandResponder&) = delete;
352 CommandResponder& operator=(const CommandResponder&) = delete;
353 CommandResponder(CommandResponder&& other) : handle_(other.handle_) { other.handle_ = nullptr; }
354 CommandResponder& operator=(CommandResponder&& other) {
355 if (this != &other) {
356 if (handle_) handle_.destroy();
357 handle_ = other.handle_;
358 other.handle_ = nullptr;
359 }
360 return *this;
361 }
362 ~CommandResponder() {
363 if (handle_) handle_.destroy();
364 }
365
369 inline bool hasMore() const override { return handle_ && !handle_.done(); }
373 catena::CommandResponse getNext() override {
374 if (hasMore()) {
375 handle_.resume();
376 handle_.promise().rethrow_if_exception();
377 }
378 return std::move(handle_.promise().responseMessage);
379 }
380
381 private:
382 std::coroutine_handle<promise_type> handle_;
383 };

440 std::function<std::unique_ptr<ICommandResponder>(const catena::Value&)> commandImpl_ = [](const catena::Value& value) -> std::unique_ptr<ICommandResponder> {
441 return std::make_unique<CommandResponder>([](const catena::Value& value) -> CommandResponder {
442 catena::CommandResponse response;
443 response.mutable_exception()->set_type("UNIMPLEMENTED");
444 response.mutable_exception()->mutable_error_message()->mutable_display_strings()->insert({"en", "Command not implemented"});
445 co_return response;
446 }(value));
