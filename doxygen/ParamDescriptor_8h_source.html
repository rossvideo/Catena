316 inline CommandResponder get_return_object() {

323 inline std::suspend_always initial_suspend() { return {}; }
327 inline std::suspend_always final_suspend() noexcept { return {}; }
332 inline std::suspend_always yield_value(st2138::CommandResponse& component) {
333 responseMessage = component;

335 }
339 inline void return_value(st2138::CommandResponse component) { this->responseMessage = component; }
343 inline void unhandled_exception() {
344 exception_ = std::current_exception();
345 }
349 inline void rethrow_if_exception() {
350 if (exception_) std::rethrow_exception(exception_);
351 }
352
353 st2138::CommandResponse responseMessage{};
354 std::exception_ptr exception_;
355 };
356
357 CommandResponder(std::coroutine_handle<promise_type> h) : handle_(h) {}
358 CommandResponder(const CommandResponder&) = delete;
359 CommandResponder& operator=(const CommandResponder&) = delete;
360 CommandResponder(CommandResponder&& other) : handle_(other.handle_) { other.handle_ = nullptr; }
361 CommandResponder& operator=(CommandResponder&& other) {
362 if (this != &other) {
363 if (handle_) handle_.destroy();
364 handle_ = other.handle_;
365 other.handle_ = nullptr;
366 }
367 return *this;
368 }
369 ~CommandResponder() {
370 if (handle_) handle_.destroy();
371 }
372
376 inline bool hasMore() const override { return handle_ && !handle_.done(); }
380 st2138::CommandResponse getNext() override {
381 if (hasMore()) {
382 handle_.resume();
383 handle_.promise().rethrow_if_exception();
384 }
385 return std::move(handle_.promise().responseMessage);
386 }
387
388 private:
389 std::coroutine_handle<promise_type> handle_;
390 };

400 void defineCommand(std::function<std::unique_ptr<ICommandResponder>(const st2138::Value&, const bool)> commandImpl) override {
401 if (!isCommand_) {
402 throw std::runtime_error("Cannot define a command on a non-command parameter");
403 }
404 commandImpl_ = commandImpl;
405 }

419 std::unique_ptr<ICommandResponder> executeCommand(const st2138::Value& value, const bool respond, catena::exception_with_status& rc, const IAuthorizer& authz) override {
420 std::unique_ptr<ICommandResponder> responder = nullptr;
421 // Make sure the client has write access.
422 if (!authz.writeAuthz(*this)) {
423 rc = catena::exception_with_status("Not authorized to execute command " + oid_, StatusCode::PERMISSION_DENIED);
424 } else {
425 responder = commandImpl_(value, respond);
426 }
427 return responder;
428 }
429
433 inline bool isCommand() const override { return isCommand_; }

460 // default command implementation
461 std::function<std::unique_ptr<ICommandResponder>(const st2138::Value&, const bool)> commandImpl_ = [](const st2138::Value& value, const bool respond) -> std::unique_ptr<ICommandResponder> {
462 return std::make_unique<CommandResponder>([](const st2138::Value& value) -> CommandResponder {
463 st2138::CommandResponse response;
464 response.mutable_exception()->set_type("UNIMPLEMENTED");
465 response.mutable_exception()->mutable_error_message()->mutable_display_strings()->insert({"en", "Command not implemented"});
466 co_return response;
467 }(value));
468 };
