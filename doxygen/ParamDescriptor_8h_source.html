284 struct promise_type {
288 inline CommandResponder get_return_object() {
289 return CommandResponder(std::coroutine_handle<promise_type>::from_promise(*this));
290 }
295 inline std::suspend_always initial_suspend() { return {}; }
299 inline std::suspend_always final_suspend() noexcept { return {}; }
304 inline std::suspend_always yield_value(catena::CommandResponse& component) {
305 responseMessage = component;
306 return {};
307 }
311 inline void return_value(catena::CommandResponse component) { this->responseMessage = component; }
315 inline void unhandled_exception() {
316 exception_ = std::current_exception();
317 }
321 inline void rethrow_if_exception() {
322 if (exception_) std::rethrow_exception(exception_);
323 }
324
325 catena::CommandResponse responseMessage{};
326 std::exception_ptr exception_;
327 };
328
329 CommandResponder(std::coroutine_handle<promise_type> h) : handle_(h) {}
330 CommandResponder(const CommandResponder&) = delete;
331 CommandResponder& operator=(const CommandResponder&) = delete;
332 CommandResponder(CommandResponder&& other) : handle_(other.handle_) { other.handle_ = nullptr; }
333 CommandResponder& operator=(CommandResponder&& other) {
334 if (this != &other) {
335 if (handle_) handle_.destroy();
336 handle_ = other.handle_;
337 other.handle_ = nullptr;
338 }
339 return *this;
340 }
341 ~CommandResponder() {
342 if (handle_) handle_.destroy();
343 }
344
348 inline bool hasMore() const override { return handle_ && !handle_.done(); }
352 catena::CommandResponse getNext() override {
353 if (hasMore()) {
354 handle_.resume();
355 handle_.promise().rethrow_if_exception();
356 }
357 return std::move(handle_.promise().responseMessage);
358 }
359
360 private:
361 std::coroutine_handle<promise_type> handle_;
362 };

419 return std::make_unique<CommandResponder>([value]() -> CommandResponder {
420 catena::CommandResponse response;
421 response.mutable_exception()->set_type("UNIMPLEMENTED");

424 }());
