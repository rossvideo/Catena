349 CommandResponder(std::coroutine_handle<promise_type> h) : handle_(h) {}
350 CommandResponder(const CommandResponder&) = delete;
351 CommandResponder& operator=(const CommandResponder&) = delete;
352 CommandResponder(CommandResponder&& other) : handle_(other.handle_) { other.handle_ = nullptr; }
353 CommandResponder& operator=(CommandResponder&& other) {
354 if (this != &other) {
355 if (handle_) handle_.destroy();
356 handle_ = other.handle_;
357 other.handle_ = nullptr;
358 }
359 return *this;
360 }
361 ~CommandResponder() {
362 if (handle_) handle_.destroy();
363 }
364
368 inline bool hasMore() const override { return handle_ && !handle_.done(); }
372 catena::CommandResponse getNext() override {
373 if (hasMore()) {
374 handle_.resume();
375 handle_.promise().rethrow_if_exception();
376 }
377 return std::move(handle_.promise().responseMessage);
378 }
379
380 private:
381 std::coroutine_handle<promise_type> handle_;
382 };

411 std::unique_ptr<ICommandResponder> executeCommand(const catena::Value& value, const bool respond, catena::exception_with_status& rc, const IAuthorizer& authz) override {
412 std::unique_ptr<ICommandResponder> responder = nullptr;
413 // Make sure the client has write access.
414 if (!authz.writeAuthz(*this)) {
415 rc = catena::exception_with_status("Not authorized to execute command " + oid_, StatusCode::PERMISSION_DENIED);
416 } else {
417 responder = commandImpl_(value, respond);
418 }
419 return responder;
420 }
421
425 inline bool isCommand() const override { return isCommand_; }

451 // default command implementation
452 std::function<std::unique_ptr<ICommandResponder>(const catena::Value&, const bool)> commandImpl_ = [](const catena::Value& value, const bool respond) -> std::unique_ptr<ICommandResponder> {
453 return std::make_unique<CommandResponder>([](const catena::Value& value) -> CommandResponder {
454 catena::CommandResponse response;
455 response.mutable_exception()->set_type("UNIMPLEMENTED");
456 response.mutable_exception()->mutable_error_message()->mutable_display_strings()->insert({"en", "Command not implemented"});
457 co_return response;
458 }(value));
459 };
