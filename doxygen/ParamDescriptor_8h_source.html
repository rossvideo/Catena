325 inline void unhandled_exception() {
326 exception_ = std::current_exception();

335 catena::CommandResponse responseMessage{};

337 };

339 CommandResponder(std::coroutine_handle<promise_type> h) : handle_(h) {}
340 CommandResponder(const CommandResponder&) = delete;
341 CommandResponder& operator=(const CommandResponder&) = delete;
342 CommandResponder(CommandResponder&& other) : handle_(other.handle_) { other.handle_ = nullptr; }
343 CommandResponder& operator=(CommandResponder&& other) {
344 if (this != &other) {
345 if (handle_) handle_.destroy();
346 handle_ = other.handle_;
347 other.handle_ = nullptr;
348 }
349 return *this;
350 }
351 ~CommandResponder() {
352 if (handle_) handle_.destroy();
353 }

358 inline bool hasMore() const override { return handle_ && !handle_.done(); }
362 catena::CommandResponse getNext() override {
363 if (hasMore()) {
364 handle_.resume();
365 handle_.promise().rethrow_if_exception();
366 }
367 return std::move(handle_.promise().responseMessage);
368 }
369
370 private:
371 std::coroutine_handle<promise_type> handle_;
372 };
