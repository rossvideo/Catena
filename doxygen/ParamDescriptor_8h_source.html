282 class CommandResponder : public ICommandResponder {
283 public:
284 struct promise_type; // forward declaration
285
286 private:
287 // The coroutine handle is a pointer to the coroutine state
288 using handle_type = std::coroutine_handle<promise_type>;
289 handle_type handle_;
290
291 public:
295 struct promise_type {
299 inline CommandResponder get_return_object() {
300 return CommandResponder(handle_type::from_promise(*this));
301 }
306 inline std::suspend_always initial_suspend() { return {}; }
310 inline std::suspend_always final_suspend() noexcept { return {}; }
315 inline std::suspend_always yield_value(catena::CommandResponse& component) {
316 responseMessage = component;
317 return {};
318 }
322 inline void return_value(catena::CommandResponse component) { this->responseMessage = component; }
326 inline void unhandled_exception() {
327 exception_ = std::current_exception();
328 }
332 inline void rethrow_if_exception() {
333 if (exception_) std::rethrow_exception(exception_);
334 }
335
336 catena::CommandResponse responseMessage{};
337 std::exception_ptr exception_;
338 };
339
340 CommandResponder(handle_type h) : handle_(h) {}
341 CommandResponder(const CommandResponder&) = delete;
342 CommandResponder& operator=(const CommandResponder&) = delete;
343 CommandResponder(CommandResponder&& other) : handle_(other.handle_) { other.handle_ = nullptr; }
344 CommandResponder& operator=(CommandResponder&& other) {
345 if (this != &other) {
346 if (handle_) handle_.destroy();
347 handle_ = other.handle_;
348 other.handle_ = nullptr;
349 }
350 return *this;
351 }
352 ~CommandResponder() {
353 if (handle_) handle_.destroy();
354 }
355
359 inline bool hasMore() const override { return handle_ && !handle_.done(); }
363 catena::CommandResponse getNext() override {
364 if (hasMore()) {
365 handle_.resume();
366 handle_.promise().rethrow_if_exception();
367 }
368 return std::move(handle_.promise().responseMessage);
369 }
370 };

427 return std::make_unique<CommandResponder>([value]() -> CommandResponder {
428 catena::CommandResponse response;
429 response.mutable_exception()->set_type("UNIMPLEMENTED");
430 response.mutable_exception()->mutable_error_message()->mutable_display_strings()->insert({"en", "Command not implemented"});
431 co_return response;
432 }());
433 };
